      setUser(data.user);

    </AuthContext.Provider>
  );

// ✅ src/context/AuthContext.jsx
import { createContext, useContext, useState, useEffect } from "react";
import { loginUser, registerUser } from "../utils/api";

// Utility to safely parse JSON with fallback
const safeParse = (key, fallback = null) => {
  try {
    const item = localStorage.getItem(key);
    return item ? JSON.parse(item) : fallback;
  } catch {
    return fallback;
  }
};

const AuthContext = createContext({
  user: null,
  token: null,
  login: async () => {},
  register: async () => {},
  logout: () => {},
});

export const AuthProvider = ({ children }) => {
  const [user, setUser] = useState(() => safeParse("user", null));
  const [token, setToken] = useState(() => safeParse("token", null));

  const login = async (credentials) => {
    const { data } = await loginUser(credentials);
    if (data?.token) {
      localStorage.setItem("token", JSON.stringify(data.token));
      setToken(data.token);
    }
    if (data?.user) {
      localStorage.setItem("user", JSON.stringify(data.user));
      setUser(data.user);
    }
  };

  const register = async (credentials) => {
    const { data } = await registerUser(credentials);
    if (data?.token) {
      localStorage.setItem("token", JSON.stringify(data.token));
      setToken(data.token);
    }
    if (data?.user) {
      localStorage.setItem("user", JSON.stringify(data.user));
      setUser(data.user);
    }
  };

  const logout = () => {
    localStorage.removeItem("user");
    localStorage.removeItem("token");
    setUser(null);
    setToken(null);
  };

  useEffect(() => {
    // Ensure state stays in sync with localStorage changes (e.g., multi-tab)
    const handleStorage = (event) => {
      if (event.key === "user") setUser(safeParse("user", null));
      if (event.key === "token") setToken(safeParse("token", null));
    };
    window.addEventListener("storage", handleStorage);
    return () => window.removeEventListener("storage", handleStorage);
  }, []);

  return (
    <AuthContext.Provider value={{ user, token, login, register, logout }}>
      {children}
    </AuthContext.Provider>
  );
};

// ✅ Safe hook for accessing AuthContext
export const useAuth = () => {
  const context = useContext(AuthContext);
  if (context === undefined || context === null) {
    throw new Error("useAuth must be used within an AuthProvider");
  }
  return context;
};
