Below is a **React implementation** meeting these requirements. Paste this into your `App.js` (or equivalent) and run.

```jsx
import React, { useState } from 'react';

function CounterOne({ onChange }) {
  const [count, setCount] = useState(0);

  const update = (newCount) => {
    setCount(newCount);
    onChange(newCount);
  };

  return (
    <div style={{ border: '1px solid #ccc', padding: '1rem', marginBottom: '1rem' }}>
      <h2>Counter One</h2>
      <p>Current Count: {count}</p>
      <button onClick={() => update(count + 1)}>Increment</button>
      <button onClick={() => update(count - 1)}>Decrement</button>
      <button onClick={() => update(count + 5)}>Add 5</button>
      <button onClick={() => update(0)}>Reset</button>
      {/* Example extra functions */}
      <button onClick={() => update(count * 2)}>×2</button>
      <button onClick={() => update(Math.floor(Math.random() * 100))}>
        Random
      </button>
    </div>
  );
}

function CounterTwo({ step }) {
  const [count, setCount] = useState(0);

  return (
    <div style={{ border: '1px solid #ccc', padding: '1rem' }}>
      <h2>Counter Two</h2>
      <p>Current Count: {count}</p>
      <button onClick={() => setCount(count + step)}>Add {step}</button>
      <button onClick={() => setCount(count - step)}>Subtract {step}</button>
      <button onClick={() => setCount(0)}>Reset</button>
    </div>
  );
}

export default function App() {
  const [stepSize, setStepSize] = useState(0);

  return (
    <div style={{ maxWidth: '400px', margin: '2rem auto', fontFamily: 'sans-serif' }}>
      <CounterOne onChange={setStepSize} />
      <CounterTwo step={stepSize} />
    </div>
  );
}
```
