# 1) Module: `SafeApp` (safety toggles that always restore)
```vba
'===== Module: SafeApp =====
Option Explicit

Private mPrevScreenUpdating As Boolean
Private mPrevEnableEvents As Boolean
Private mPrevDisplayAlerts As Boolean
Private mPrevCalcMode As XlCalculation
Private mDepth As Long

Public Sub Begin(ByVal contextName As String)
    ' Nested-safe: only capture previous states at outermost Begin
    If mDepth = 0 Then
        mPrevScreenUpdating = Application.ScreenUpdating
        mPrevEnableEvents = Application.EnableEvents
        mPrevDisplayAlerts = Application.DisplayAlerts
        mPrevCalcMode = Application.Calculation

        Application.ScreenUpdating = False
        Application.EnableEvents = False
        Application.DisplayAlerts = False
        Application.Calculation = xlCalculationManual
    End If
    mDepth = mDepth + 1
    Logger.LogEvent "SAFE_BEGIN", contextName
End Sub

Public Sub Finish(Optional ByVal contextName As String = "")
    If mDepth > 0 Then mDepth = mDepth - 1
    If mDepth = 0 Then
        Application.Calculation = mPrevCalcMode
        Application.DisplayAlerts = mPrevDisplayAlerts
        Application.EnableEvents = mPrevEnableEvents
        Application.ScreenUpdating = mPrevScreenUpdating
    End If
    If Len(contextName) > 0 Then Logger.LogEvent "SAFE_FINISH", contextName
End Sub

Public Sub PanicRestore()
    ' Call this if something breaks mid-run to restore Excel state
    Application.Calculation = xlCalculationAutomatic
    Application.DisplayAlerts = True
    Application.EnableEvents = True
    Application.ScreenUpdating = True
End Sub
```

# 2) Module: `Logger` (writes to a `DebugLog` sheet)
```vba
'===== Module: Logger =====
Option Explicit

Private Const LOG_SHEET_NAME As String = "DebugLog"

Private Function EnsureLogSheet() As Worksheet
    Dim ws As Worksheet
    On Error Resume Next
    Set ws = ThisWorkbook.Worksheets(LOG_SHEET_NAME)
    On Error GoTo 0
    If ws Is Nothing Then
        Set ws = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Sheets(ThisWorkbook.Sheets.Count))
        ws.Name = LOG_SHEET_NAME
        With ws.Range("A1:H1")
            .Value = Array("Timestamp", "Level", "Event", "Proc", "Workbook", "Sheet", "Cell", "Note")
        End With
    End If
    Set EnsureLogSheet = ws
End Function

Public Sub Log(ByVal level As String, ByVal evt As String, ByVal proc As String, _
               Optional ByVal note As String = "")
    Dim ws As Worksheet, r As Long
    Set ws = EnsureLogSheet()
    r = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row + 1
    ws.Cells(r, 1).Value = Now
    ws.Cells(r, 2).Value = level
    ws.Cells(r, 3).Value = evt
    ws.Cells(r, 4).Value = proc
    ws.Cells(r, 5).Value = OnWorkbookName()
    ws.Cells(r, 6).Value = OnSheetName()
    ws.Cells(r, 7).Value = OnCellAddress()
    ws.Cells(r, 8).Value = note
End Sub

Public Sub LogEvent(ByVal evt As String, _
                    Optional ByVal note As String = "", _
                    Optional ByVal proc As String = "")
    Log "INFO", evt, proc, note
End Sub

Public Sub LogWarn(ByVal evt As String, Optional ByVal note As String = "", Optional ByVal proc As String = "")
    Log "WARN", evt, proc, note
End Sub

Public Sub LogError(ByVal evt As String, Optional ByVal note As String = "", Optional ByVal proc As String = "")
    Log "ERROR", evt, proc, note
End Sub

Private Function OnWorkbookName() As String
    On Error Resume Next
    OnWorkbookName = ActiveWorkbook.Name
End Function

Private Function OnSheetName() As String
    On Error Resume Next
    If Not ActiveSheet Is Nothing Then OnSheetName = ActiveSheet.Name
End Function

Private Function OnCellAddress() As String
    On Error Resume Next
    If Not Application.Caller Is Nothing Then
        OnCellAddress = Application.Caller.Address
    ElseIf Not Selection Is Nothing Then
        OnCellAddress = Selection(1).Address
    End If
End Function
```

# 3) Module: `ErrHandler` (centralized error capture)
```vba
'===== Module: ErrHandler =====
Option Explicit

Public Sub Handle(ByVal procName As String, Optional ByVal extra As String = "")
    Dim msg As String
    msg = "Err " & Err.Number & " — " & Err.Description & _
          " | Source: " & Err.Source & _
          IIf(Len(extra) > 0, " | Extra: " & extra, "")
    Logger.LogError "EXCEPTION", msg, procName
End Sub
```

# 4) Module: `Utils` (defensive helpers)
```vba
'===== Module: Utils =====
Option Explicit

Public Function SheetExists(ByVal wb As Workbook, ByVal sheetName As String) As Boolean
    Dim ws As Worksheet
    On Error Resume Next
    Set ws = wb.Worksheets(sheetName)
    SheetExists = Not ws Is Nothing
    On Error GoTo 0
End Function

Public Function EnsureSheet(ByVal wb As Workbook, ByVal sheetName As String) As Worksheet
    If SheetExists(wb, sheetName) Then
        Set EnsureSheet = wb.Worksheets(sheetName)
    Else
        Set EnsureSheet = wb.Worksheets.Add(After:=wb.Sheets(wb.Sheets.Count))
        EnsureSheet.Name = sheetName
    End If
End Function

Public Function LastRow(ByVal ws As Worksheet, Optional ByVal col As Long = 1) As Long
    Dim lr As Long
    lr = ws.Cells(ws.Rows.Count, col).End(xlUp).Row
    If lr < 1 Then lr = 1
    LastRow = lr
End Function

Public Function LastCol(ByVal ws As Worksheet, Optional ByVal row As Long = 1) As Long
    Dim lc As Long
    lc = ws.Cells(row, ws.Columns.Count).End(xlToLeft).Column
    If lc < 1 Then lc = 1
    LastCol = lc
End Function

Public Function GetRangeOrNothing(ByVal ws As Worksheet, ByVal addr As String) As Range
    On Error Resume Next
    Set GetRangeOrNothing = ws.Range(addr)
    On Error GoTo 0
End Function

Public Function FindText(ByVal ws As Worksheet, ByVal whatText As String, _
                         Optional ByVal inCol As Long = 1) As Range
    Dim rng As Range
    Set rng = ws.Columns(inCol).Find(What:=whatText, LookIn:=xlValues, LookAt:=xlWhole, _
                                     SearchOrder:=xlByRows, SearchDirection:=xlNext, MatchCase:=False)
    If Not rng Is Nothing Then Set FindText = rng
End Function

Public Sub SafeClearRows(ByVal ws As Worksheet, ByVal firstRow As Long, ByVal lastRow As Long)
    Dim r As Long
    If lastRow < firstRow Then Exit Sub
    For r = lastRow To firstRow Step -1
        ws.Rows(r).Delete
    Next r
End Sub
```

# 6) Common causes of intermittent VBA errors (and quick fixes)
- Unqualified references (`Rows.Delete`, `Cells`, `Range`) act on `ActiveSheet`. Fix by qualifying (`ws.Rows`, `ws.Cells`, `ws.Range`).
- Deleting while looping top-down skips rows. Fix by deleting bottom-up (see `SafeClearRows`).
- `.Find` returning `Nothing`. Always test before using result.
- Missing/renamed sheets. Guard with `SheetExists`.
- `UsedRange` stale after heavy edits. Force reset by saving workbook or toggling visibility; better: avoid relying on `UsedRange` for bounds—use `LastRow/LastCol`.
- Events/ScreenUpdating/Calculation not restored after a crash. Fix with `SafeApp` and centralized `ErrHandler`.
- Mixed data types (e.g., `Integer` overflow). Use `Long` for row/column variables.
- Hidden/filtered rows interfering with loops. Prefer working with a fully visible state or use `SpecialCells(xlCellTypeVisible)` carefully.
