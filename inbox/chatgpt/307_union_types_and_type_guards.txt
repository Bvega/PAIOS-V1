**Critical Thinking Answer**  
Union types and type guards make your code safer and more maintainable by ensuring that every possible input shape is explicitly handled. With a union type, the compiler knows that `input` can only be `string[]` or `number[]`, so you can’t accidentally pass in an unsupported type. Type guards then let you narrow down which branch of logic to execute, eliminating runtime errors due to unexpected types. This approach mitigates risks like calling string methods on numbers or vice versa, and makes future refactoring easier because all cases are centrally defined and checked by the type system.

    // Type guard to check if input is string[]

ASSISTANT:
```ts
// dataProcessor.ts

// DataArray can be either an array of strings or an array of numbers
type DataArray = string[] | number[];

/**
 * Takes a string[] or number[] and returns a comma-separated string:
 *  - string[] → uppercase each item
 *  - number[] → two-decimal rounding
 *
 * @param input – array of strings or numbers
 * @returns formatted comma-separated string ('' if input is empty)
 */
export function processData(input: DataArray): string {
    // handle empty array
    if (input.length === 0) {
        return '';
    }

    // type guard: first element is a string ⇒ treat entire array as string[]
    if (typeof input[0] === 'string') {
        return (input as string[])
            .map(item => item.toUpperCase())    // uppercase each string
            .join(', ');                        // join with commas
    }

    // otherwise treat as number[]
    return (input as number[])
        .map(num => num.toFixed(2))            // round to two decimals
        .join(', ');                           // join with commas
}
```
