### 1. **Using a Script (Python Example):**

def generate_prompt(style, subject, hair, lighting, shadows, expression, contrast, texture, design, shot_type, angle, mood, atmosphere, emphasis):
    prompt = f"""
    An elegant {style} of a {subject} with {hair} that curls gracefully around her. The lighting is {lighting}, casting {shadows} that highlight her {expression}. 
    The artwork features {contrast} with {texture}, capturing a blend of {design}. The scene is framed as a {shot_type}, viewed from a {angle} to convey a sense of {mood}. 
    The mood is {atmosphere}, with an air of {emphasis}.
    """
    return prompt.strip()

# Example of calling the function with user input
style = input("Choose a style (e.g., monochromatic digital portrait): ")

shadows = input("Describe the shadows (e.g., deep shadows): ")

contrast = input("Describe the contrast (e.g., high contrast): ")
texture = input("Describe the texture (e.g., smooth, polished textures): ")
design = input("Describe the design style (e.g., realism and stylized design): ")
shot_type = input("Choose a shot type (e.g., close-up head-and-shoulders shot): ")
angle = input("Choose an angle (e.g., slightly low angle): ")
mood = input("Describe the mood (e.g., grace and empowerment): ")
atmosphere = input("Describe the atmosphere (e.g., sophisticated and calm): ")
emphasis = input("Choose an emphasis (e.g., timeless beauty): ")

print(generate_prompt(style, subject, hair, lighting, shadows, expression, contrast, texture, design, shot_type, angle, mood, atmosphere, emphasis))
