```python
import tkinter as tk
from tkinter import filedialog, messagebox
import os
import re
import io
from PyPDF2 import PdfReader, PdfWriter
from docx import Document

def anonymize_pdf(file_path, patterns):
    reader = PdfReader(file_path)
    writer = PdfWriter()

    for page_num, page in enumerate(reader.pages):
        text = page.extract_text()
        if text:
            for pattern in patterns:
                text = re.sub(pattern, '[REDACTED]', text, flags=re.IGNORECASE)
            packet = io.BytesIO()
            temp_writer = PdfWriter()
            temp_writer.add_page(page)
            temp_writer.add_metadata(reader.metadata)
            temp_writer.write(packet)
            packet.seek(0)
            writer.add_page(PdfReader(packet).pages[0])
        else:
            writer.add_page(page)

    output_path = os.path.splitext(file_path)[0] + '_anonymized.pdf'
    with open(output_path, 'wb') as out_file:
        writer.write(out_file)

def anonymize_docx(file_path, patterns):
    doc = Document(file_path)

    for para in doc.paragraphs:
        for pattern in patterns:
            if re.search(pattern, para.text, flags=re.IGNORECASE):
                para.text = re.sub(pattern, '[REDACTED]', para.text, flags=re.IGNORECASE)

    for table in doc.tables:
        for row in table.rows:
            for cell in row.cells:
                for pattern in patterns:
                    if re.search(pattern, cell.text, flags=re.IGNORECASE):
                        cell.text = re.sub(pattern, '[REDACTED]', cell.text, flags=re.IGNORECASE)

    output_path = os.path.splitext(file_path)[0] + '_anonymized.docx'
    doc.save(output_path)

def select_file():
    file_path = filedialog.askopenfilename(
        filetypes=[('PDF and Word Documents', '*.pdf *.docx')])
    if file_path:
        patterns = [entry.get() for entry in pattern_entries if entry.get()]
        if not patterns:
            messagebox.showwarning('No Patterns', 'Please enter at least one pattern to anonymize.')
            return
        try:
            if file_path.endswith('.pdf'):
                anonymize_pdf(file_path, patterns)
            elif file_path.endswith('.docx'):
                anonymize_docx(file_path, patterns)
            messagebox.showinfo('Success', 'File anonymized successfully!')
        except Exception as e:
            messagebox.showerror('Error', f'An error occurred: {e}')

def add_pattern_entry():
    entry = tk.Entry(pattern_frame, width=50)
    entry.pack(pady=2)
    pattern_entries.append(entry)

def main():
    global pattern_entries, pattern_frame

    root = tk.Tk()
    root.title('Contract Anonymizer')

    instruction = tk.Label(root, text='Enter patterns to anonymize (e.g., names, addresses):')
    instruction.pack(pady=10)

    pattern_frame = tk.Frame(root)
    pattern_frame.pack()

    pattern_entries = []
    add_pattern_entry()

    add_pattern_btn = tk.Button(root, text='Add Another Pattern', command=add_pattern_entry)
    add_pattern_btn.pack(pady=5)

    select_file_btn = tk.Button(root, text='Select Contract File', command=select_file)
    select_file_btn.pack(pady=20)

    root.mainloop()

if __name__ == '__main__':
    main()
```

```python
import tkinter as tk
from tkinter import filedialog, messagebox
import os
import re
import io
from PyPDF2 import PdfReader, PdfWriter
from docx import Document

def anonymize_pdf(input_path, output_path, patterns):
    reader = PdfReader(input_path)
    writer = PdfWriter()

    for page_num, page in enumerate(reader.pages):
        text = page.extract_text()
        if text:
            for pattern in patterns:
                text = re.sub(pattern, '[REDACTED]', text, flags=re.IGNORECASE)
            packet = io.BytesIO()
            temp_writer = PdfWriter()
            temp_writer.add_page(page)
            temp_writer.add_metadata(reader.metadata)
            temp_writer.write(packet)
            packet.seek(0)
            writer.add_page(PdfReader(packet).pages[0])
        else:
            writer.add_page(page)

    with open(output_path, 'wb') as out_file:
        writer.write(out_file)

def anonymize_docx(input_path, output_path, patterns):
    doc = Document(input_path)

    for para in doc.paragraphs:
        for pattern in patterns:
            if re.search(pattern, para.text, flags=re.IGNORECASE):
                para.text = re.sub(pattern, '[REDACTED]', para.text, flags=re.IGNORECASE)

    for table in doc.tables:
        for row in table.rows:
            for cell in row.cells:
                for pattern in patterns:
                    if re.search(pattern, cell.text, flags=re.IGNORECASE):
                        cell.text = re.sub(pattern, '[REDACTED]', cell.text, flags=re.IGNORECASE)

    doc.save(output_path)

def select_input_file():
    input_path = filedialog.askopenfilename(
        filetypes=[('PDF and Word Documents', '*.pdf *.docx')])
    if input_path:
        input_entry.delete(0, tk.END)
        input_entry.insert(0, input_path)

def select_output_file():
    output_path = filedialog.asksaveasfilename(
        defaultextension=".pdf" if input_entry.get().endswith('.pdf') else '.docx',
        filetypes=[('PDF Files', '*.pdf'), ('Word Documents', '*.docx')])
    if output_path:
        output_entry.delete(0, tk.END)
        output_entry.insert(0, output_path)

def anonymize_file():
    input_path = input_entry.get()
    output_path = output_entry.get()

    if not input_path or not os.path.isfile(input_path):
        messagebox.showwarning('Invalid Input', 'Please select a valid input file.')
        return

    if not output_path:
        messagebox.showwarning('Invalid Output', 'Please select a valid output path.')
        return

    patterns = [entry.get() for entry in pattern_entries if entry.get()]
    if not patterns:
        messagebox.showwarning('No Patterns', 'Please enter at least one pattern to anonymize.')
        return

    try:
        if input_path.endswith('.pdf'):
            anonymize_pdf(input_path, output_path, patterns)
        elif input_path.endswith('.docx'):
            anonymize_docx(input_path, output_path, patterns)
        else:
            messagebox.showerror('Unsupported Format', 'The selected file format is not supported.')
            return
        messagebox.showinfo('Success', f'File anonymized successfully!\nSaved to: {output_path}')
    except Exception as e:
        messagebox.showerror('Error', f'An error occurred: {e}')

def add_pattern_entry():
    entry = tk.Entry(pattern_frame, width=50)
    entry.pack(pady=2)
    pattern_entries.append(entry)

def main():
    global pattern_entries, pattern_frame, input_entry, output_entry

    root = tk.Tk()
    root.title('Contract Anonymizer')

    # Input File Selection
    input_label = tk.Label(root, text='Select Input File:')
    input_label.pack(pady=(10, 0))

    input_frame = tk.Frame(root)
    input_frame.pack()

    input_entry = tk.Entry(input_frame, width=50)
    input_entry.pack(side=tk.LEFT, padx=(0, 5))

    input_browse_btn = tk.Button(input_frame, text='Browse', command=select_input_file)
    input_browse_btn.pack(side=tk.LEFT)

    # Output File Selection
    output_label = tk.Label(root, text='Select Output File:')
    output_label.pack(pady=(10, 0))

    output_frame = tk.Frame(root)
    output_frame.pack()

    output_entry = tk.Entry(output_frame, width=50)
    output_entry.pack(side=tk.LEFT, padx=(0, 5))

    output_browse_btn = tk.Button(output_frame, text='Browse', command=select_output_file)
    output_browse_btn.pack(side=tk.LEFT)

    # Pattern Entries
    instruction = tk.Label(root, text='Enter patterns to anonymize (e.g., names, addresses):')
    instruction.pack(pady=10)

    pattern_frame = tk.Frame(root)
    pattern_frame.pack()

    pattern_entries = []
    add_pattern_entry()

    add_pattern_btn = tk.Button(root, text='Add Another Pattern', command=add_pattern_entry)
    add_pattern_btn.pack(pady=5)

    # Anonymize Button
    anonymize_btn = tk.Button(root, text='Anonymize File', command=anonymize_file)
    anonymize_btn.pack(pady=20)

    root.mainloop()

    main()
```

Copy and paste the following code into `contract_anonymizer.py`:

```python
import tkinter as tk
from tkinter import filedialog, messagebox
import os
import re
import io
from PyPDF2 import PdfReader, PdfWriter
from docx import Document

def anonymize_pdf(input_path, output_path, patterns):
    reader = PdfReader(input_path)
    writer = PdfWriter()

    for page_num, page in enumerate(reader.pages):
        text = page.extract_text()
        if text:
            for pattern in patterns:
                text = re.sub(pattern, '[REDACTED]', text, flags=re.IGNORECASE)
            packet = io.BytesIO()
            temp_writer = PdfWriter()
            temp_writer.add_page(page)
            temp_writer.add_metadata(reader.metadata)
            temp_writer.write(packet)
            packet.seek(0)
            writer.add_page(PdfReader(packet).pages[0])
        else:
            writer.add_page(page)

    with open(output_path, 'wb') as out_file:
        writer.write(out_file)

def anonymize_docx(input_path, output_path, patterns):
    doc = Document(input_path)

    for para in doc.paragraphs:
        for pattern in patterns:
            if re.search(pattern, para.text, flags=re.IGNORECASE):
                para.text = re.sub(pattern, '[REDACTED]', para.text, flags=re.IGNORECASE)

    for table in doc.tables:
        for row in table.rows:
            for cell in row.cells:
                for pattern in patterns:
                    if re.search(pattern, cell.text, flags=re.IGNORECASE):
                        cell.text = re.sub(pattern, '[REDACTED]', cell.text, flags=re.IGNORECASE)

    doc.save(output_path)

def select_input_file():
    input_path = filedialog.askopenfilename(
        filetypes=[('PDF and Word Documents', '*.pdf *.docx')])
    if input_path:
        input_entry.delete(0, tk.END)
        input_entry.insert(0, input_path)

def select_output_file():
    output_path = filedialog.asksaveasfilename(
        defaultextension=".pdf" if input_entry.get().endswith('.pdf') else '.docx',
        filetypes=[('PDF Files', '*.pdf'), ('Word Documents', '*.docx')])
    if output_path:
        output_entry.delete(0, tk.END)
        output_entry.insert(0, output_path)

def anonymize_file():
    input_path = input_entry.get()
    output_path = output_entry.get()

    if not input_path or not os.path.isfile(input_path):
        messagebox.showwarning('Invalid Input', 'Please select a valid input file.')
        return

    if not output_path:
        messagebox.showwarning('Invalid Output', 'Please select a valid output path.')
        return

    patterns = [entry.get() for entry in pattern_entries if entry.get()]
    if not patterns:
        messagebox.showwarning('No Patterns', 'Please enter at least one pattern to anonymize.')
        return

    try:
        if input_path.endswith('.pdf'):
            anonymize_pdf(input_path, output_path, patterns)
        elif input_path.endswith('.docx'):
            anonymize_docx(input_path, output_path, patterns)
        else:
            messagebox.showerror('Unsupported Format', 'The selected file format is not supported.')
            return
        messagebox.showinfo('Success', f'File anonymized successfully!\nSaved to: {output_path}')
    except Exception as e:
        messagebox.showerror('Error', f'An error occurred: {e}')

def add_pattern_entry():
    entry = tk.Entry(pattern_frame, width=50)
    entry.pack(pady=2)
    pattern_entries.append(entry)

def main():
    global pattern_entries, pattern_frame, input_entry, output_entry

    root = tk.Tk()
    root.title('Contract Anonymizer')

    # Input File Selection
    input_label = tk.Label(root, text='Select Input File:')
    input_label.pack(pady=(10, 0))

    input_frame = tk.Frame(root)
    input_frame.pack()

    input_entry = tk.Entry(input_frame, width=50)
    input_entry.pack(side=tk.LEFT, padx=(0, 5))

    input_browse_btn = tk.Button(input_frame, text='Browse', command=select_input_file)
    input_browse_btn.pack(side=tk.LEFT)

    # Output File Selection
    output_label = tk.Label(root, text='Select Output File:')
    output_label.pack(pady=(10, 0))

    output_frame = tk.Frame(root)
    output_frame.pack()

    output_entry = tk.Entry(output_frame, width=50)
    output_entry.pack(side=tk.LEFT, padx=(0, 5))

    output_browse_btn = tk.Button(output_frame, text='Browse', command=select_output_file)
    output_browse_btn.pack(side=tk.LEFT)

    # Pattern Entries
    instruction = tk.Label(root, text='Enter patterns to anonymize (e.g., names, addresses):')
    instruction.pack(pady=10)

    pattern_frame = tk.Frame(root)
    pattern_frame.pack()

    pattern_entries = []
    add_pattern_entry()

    add_pattern_btn = tk.Button(root, text='Add Another Pattern', command=add_pattern_entry)
    add_pattern_btn.pack(pady=5)

    # Anonymize Button
    anonymize_btn = tk.Button(root, text='Anonymize File', command=anonymize_file)
    anonymize_btn.pack(pady=20)

    root.mainloop()

if __name__ == '__main__':
    main()
```
