# parse_chat_export.py (main script)

parser_script = r'''
#!/usr/bin/env python3
"""
Parse ChatGPT data export (conversations.json) into structured knowledge for GPT Builder.

Outputs:
- out/knowledge.jsonl               : JSONL knowledge items
- out/categories/<cat>/summary.md   : Per-category markdown summary
- out/categories/<cat>/examples.md  : Few-shot examples
- out/bolivar_knowledge_bundle.zip  : Zipped bundle of the above

Usage:
  python parse_chat_export.py --input conversations.json --config config.yaml --out out
"""
import argparse
import json
import os
import re
import sys
import zipfile
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List, Tuple

try:
    import yaml
except ImportError:
    print("This script requires PyYAML. Install with: pip install pyyaml", file=sys.stderr)
    sys.exit(1)

MIN_EXAMPLE_LEN_DEFAULT = 50

def load_config(path: Path) -> Dict:
    with path.open("r", encoding="utf-8") as f:
        cfg = yaml.safe_load(f)
    if "output" not in cfg:
        cfg["output"] = {}
    if "dir" not in cfg["output"]:
        cfg["output"]["dir"] = "out"
    if "max_examples_per_context" not in cfg["output"]:
        cfg["output"]["max_examples_per_context"] = 6
    if "min_example_len" not in cfg["output"]:
        cfg["output"]["min_example_len"] = MIN_EXAMPLE_LEN_DEFAULT
    return cfg

def read_conversations(path: Path) -> List[Dict]:
    with path.open("r", encoding="utf-8") as f:
        data = json.load(f)
    # The export format typically has an array of conversations
    # Each has a title and a mapping with messages under "mapping"
    if isinstance(data, dict) and "conversations" in data:
        return data["conversations"]
    if isinstance(data, list):
        return data
    raise ValueError("Unrecognized conversations.json format")

def normalize_text(s: str) -> str:
    return (s or "").strip()

def message_pairs(convo: Dict) -> List[Tuple[str, str]]:
    """
    Extract (user, assistant) message pairs in chronological order.
    """
    pairs = []
    # Some exports use "mapping" with nodes
    if "mapping" in convo and isinstance(convo["mapping"], dict):
        # Build a list of nodes in chronological order by create_time
        nodes = [v for v in convo["mapping"].values() if isinstance(v, dict) and v.get("message")]
        nodes.sort(key=lambda n: (n["message"].get("create_time") or 0))
        last_user = None
        for n in nodes:
            msg = n["message"]
            role = msg.get("author", {}).get("role", "")
            content_parts = msg.get("content", {}).get("parts") or []
            text = "\n".join([p for p in content_parts if isinstance(p, str)])
            text = normalize_text(text)
            if not text:
                continue
            if role == "user":
                last_user = text
            elif role in ("assistant", "tool"):
                if last_user:
                    pairs.append((last_user, text))
                    last_user = None
    else:
        # Fallback: if there is a flat list of messages
        msgs = convo.get("messages") or []
        last_user = None
        for m in msgs:
            role = m.get("role") or m.get("author", {}).get("role", "")
            text = normalize_text(m.get("content") or m.get("text") or "")
            if not text:
                continue
            if role == "user":
                last_user = text
            elif role in ("assistant", "tool"):
                if last_user:
                    pairs.append((last_user, text))
                    last_user = None
    return pairs

def classify(text: str, cfg: Dict) -> str:
    """Simple keyword-based category assignment (case-insensitive)."""
    t = text.lower()
    for key, block in (cfg.get("categories") or {}).items():
        kws = [k.lower() for k in (block.get("keywords") or [])]
        if any(k in t for k in kws):
            return block.get("name") or key
    return cfg.get("default_category", "General")

def ensure_dir(p: Path):
    p.mkdir(parents=True, exist_ok=True)

def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--input", required=True, help="Path to conversations.json from ChatGPT export")
    ap.add_argument("--config", required=True, help="Path to config.yaml")
    ap.add_argument("--out", default=None, help="Output directory (overrides config.yaml output.dir)")
    args = ap.parse_args()

    cfg = load_config(Path(args.config))
    out_dir = Path(args.out) if args.out else Path(cfg["output"]["dir"])
    ensure_dir(out_dir)

    conversations = read_conversations(Path(args.input))

    # Accumulators
    category_examples: Dict[str, List[Tuple[str, str]]] = {}
    knowledge_items: List[Dict] = []

    min_len = int(cfg["output"]["min_example_len"])
    max_per = int(cfg["output"]["max_examples_per_context"])

    now_iso = datetime.now(timezone.utc).isoformat()

    for convo in conversations:
        title = normalize_text(convo.get("title") or convo.get("conversation_id") or "Untitled")
        pairs = message_pairs(convo)
        if not pairs:
            continue

        # Collate full text to classify
        full_text = (title + "\n" + "\n\n".join([u + "\n---\n" + a for u,a in pairs]))[:5000]
        cat = classify(full_text, cfg)

        # Save examples for this category
        kept = 0
        for (u,a) in pairs:
            if len(u) + len(a) < min_len:
                continue
            category_examples.setdefault(cat, [])
            if kept < max_per:
                category_examples[cat].append((u,a))
                kept += 1

        # Create a knowledge item per conversation (summary-level, with title as context)
        item = {
            "context": title,
            "subcontext": cat,
            "instructions": f"This context derives from a past conversation titled '{title}'. Prioritize actionable outputs, provide code or files as needed, mirror bilingual tone. Category: {cat}.",
            "examples": [],
            "tags": [cat],
            "links": [],
            "updated_at": now_iso
        }
        # Attach up to 3 example pairs to the item
        for (u,a) in (category_examples.get(cat, [])[:3] if category_examples.get(cat) else []):
            item["examples"].append({"user": u, "assistant": a})
        knowledge_items.append(item)

    # Write JSONL
    kj = out_dir / "knowledge.jsonl"
    with kj.open("w", encoding="utf-8") as f:
        for item in knowledge_items:
            f.write(json.dumps(item, ensure_ascii=False) + "\n")

    # Write per-category markdown
    cat_dir = out_dir / "categories"
