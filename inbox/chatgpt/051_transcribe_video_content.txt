def extract_audio_ffmpeg(video_path, wav_out):
    # Use ffmpeg CLI for reliable extraction
    cmd = [
        "ffmpeg",
        "-y",
        "-i", video_path,
        "-ac", "1",           # mono
        "-ar", "16000",       # 16 kHz
        "-vn",
        wav_out
    ]
    try:
        proc = subprocess.run(cmd, capture_output=True, text=True, timeout=180)
        return proc.returncode == 0, proc.stderr
    except Exception as e:
        return False, str(e)

def try_whisper(wav_path):
    """
    Attempt transcription with openai-whisper if installed and a model is locally cached.
    We'll try tiny/base/small in that order without triggering downloads.
    """
    try:
        import whisper  # openai-whisper
    except Exception as e:
        return None, f"whisper import failed: {e}"

    # Check cache dir for any model files
    possible_models = ["tiny", "base", "small"]
    # If models aren't cached, whisper will attempt a download which will fail.
    # So we guard by checking for model files in default cache paths.
    cache_dirs = []
    # Common cache locations
    for envvar in ["WHISPER_CACHE_DIR", "XDG_CACHE_HOME", "HOME"]:
        v = os.environ.get(envvar)
        if v:
            if envvar == "WHISPER_CACHE_DIR":
                cache_dirs.append(v)
            elif envvar == "XDG_CACHE_HOME":
                cache_dirs.append(os.path.join(v, "whisper"))
            elif envvar == "HOME":
                cache_dirs.append(os.path.join(v, ".cache", "whisper"))
    cache_dirs.append("/mnt/data/whisper")  # custom try

    def model_cached(model_name):
        for d in cache_dirs:
            if os.path.isdir(d):
                for fname in os.listdir(d):
                    if model_name in fname and (fname.endswith(".bin") or fname.endswith(".pt")):
                        return True
        return False

    chosen = None
    for m in possible_models:
        if model_cached(m):
            chosen = m
            break

    if chosen is None:
        # As a last-ditch, attempt load tiny — if it's not cached this will likely fail quickly.
        chosen = "tiny"

    try:
        model = whisper.load_model(chosen, download_root="/mnt/data/whisper", device="cpu")
        # transcribe
        result = model.transcribe(wav_path, fp16=False)
        text = result.get("text", "").strip()
        segments = result.get("segments", [])
        return {"text": text, "segments": segments, "engine": f"whisper-{chosen}"}, None
    except Exception as e:
        return None, f"whisper failed: {e}"

def try_sphinx(wav_path):
    """
    Attempt transcription with PocketSphinx if installed.
    Two strategies: direct pocketsphinx API or via speech_recognition.
    """
    # Strategy 1: pocketsphinx API
    try:
        from pocketsphinx import AudioFile  # type: ignore
        # Minimal config; relies on default model paths (if any)
        config = {
            "audio_file": wav_path
        }
        audio = AudioFile(**config)
        collected = []
        for phrase in audio:
            collected.append(str(phrase))
        text = " ".join(collected).strip()
        if text:
            return {"text": text, "segments": [], "engine": "pocketsphinx"}, None
    except Exception as e:
        sphinx_api_err = f"pocketsphinx API failed: {e}"
    else:
        sphinx_api_err = None

    # Strategy 2: SpeechRecognition wrapper
    try:
        import speech_recognition as sr  # type: ignore
        r = sr.Recognizer()
        with sr.AudioFile(wav_path) as source:
            audio = r.record(source)
        text = r.recognize_sphinx(audio)
        if text:
            return {"text": text.strip(), "segments": [], "engine": "SpeechRecognition+Sphinx"}, None
    except Exception as e:
        sr_err = f"SpeechRecognition Sphinx failed: {e}"
    else:
        sr_err = None

    return None, "; ".join([x for x in [sphinx_api_err, sr_err] if x])

def write_srt(segments, srt_out_path):
    """
    Write SRT from Whisper-like segments (with start/end + text).
    """
    try:
        if not segments:
            return False
        def fmt_ts(t):
            hr = int(t // 3600)
            t -= hr*3600
            mn = int(t // 60)
            t -= mn*60
            sc = int(t)
            ms = int((t - sc) * 1000)
            return f"{hr:02d}:{mn:02d}:{sc:02d},{ms:03d}"

        lines = []
        for i, seg in enumerate(segments, 1):
            start = seg.get("start", 0.0)
            end = seg.get("end", start + 0.5)
            txt = seg.get("text", "").strip()
            lines.append(str(i))
            lines.append(f"{fmt_ts(start)} --> {fmt_ts(end)}")
            lines.append(txt)
            lines.append("")
        with open(srt_out_path, "w", encoding="utf-8") as f:
            f.write("\n".join(lines))
        return True
    except Exception:
        return False
