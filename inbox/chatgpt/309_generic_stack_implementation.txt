### File: `stack.ts`

```typescript
// Generic Stack Class
class Stack<T> {
  private items: T[] = [];

  // Push item onto the stack
  push(item: T): void {
    this.items.push(item);
  }

  // Pop item from the top of the stack
  pop(): T | undefined {
    return this.items.pop();
  }

  // Peek at the top item without removing it
  peek(): T | undefined {
    return this.items[this.items.length - 1];
  }

  // Check if the stack is empty
  isEmpty(): boolean {
    return this.items.length === 0;
  }

  // Get the size of the stack
  size(): number {
    return this.items.length;
  }
}

// Demonstration with different data types

// Stack of numbers
const numberStack = new Stack<number>();
numberStack.push(10);
numberStack.push(20);
console.log("Peek number:", numberStack.peek()); // 20
console.log("Pop number:", numberStack.pop()); // 20

// Stack of strings
const stringStack = new Stack<string>();
stringStack.push("hello");
stringStack.push("world");
console.log("Peek string:", stringStack.peek()); // world
console.log("Pop string:", stringStack.pop()); // world
```

### Critical Thinking

#### How does using generics prevent type-related errors?

Generics allow the Stack to maintain type safety. By specifying a type `T`, you ensure that all elements pushed to the stack are of that type. TypeScript's compiler will catch any mismatches at compile time, reducing runtime errors and improving code clarity.

#### Non-generic version and its limitations

A non-generic Stack might use `any`:

```typescript
class NonGenericStack {
  private items: any[] = [];

  push(item: any): void {
    this.items.push(item);
  }

  pop(): any {
    return this.items.pop();
  }

  peek(): any {
    return this.items[this.items.length - 1];
  }
}
```

**Limitations:**
- No compile-time type checking.
- Risk of type mismatches and runtime errors.
- Less clarity for consumers of the Stack class about what types are valid.

```ts
// A generic Stack class implementation using TypeScript
// This supports any data type by using the generic type parameter <T>

export class Stack<T> {
  // Internal array to store stack items
  private items: T[] = [];

  /**
   * Adds a new item to the top of the stack
   * @param item - The item to add
   */
  push(item: T): void {
    this.items.push(item);
  }

  /**
   * Removes and returns the top item from the stack
   * Returns undefined if the stack is empty
   */
  pop(): T | undefined {
    return this.items.pop();
  }

  /**
   * Returns the top item without removing it
   * Useful for checking what's next
   */
  peek(): T | undefined {
    return this.items[this.items.length - 1];
  }

  /**
   * Checks whether the stack is empty
   */
  isEmpty(): boolean {
    return this.items.length === 0;
  }

  /**
   * Returns the number of items in the stack
   */
  size(): number {
    return this.items.length;
  }
}
```
