Lab Overview
In this lab, you will create an interactive character counter application that demonstrates state management and event handling in React. Workplace Context

**8. Implement StatsDisplay component**

*Explanation:*  
The `StatsDisplay` component will receive computed stats (character count, word count, reading time) and optionally display the reading time. We’ll structure it for clarity and add comments.

**Action:**  
Open `src/components/StatsDisplay/StatsDisplay.tsx` and add:

```tsx
// src/components/StatsDisplay/StatsDisplay.tsx
import React from 'react';
import { StatsDisplayProps } from '../../types';

/**
 * StatsDisplay component
 * Shows character count, word count, and optional reading time.
 */
export const StatsDisplay: React.FC<StatsDisplayProps> = ({
  stats,
  showReadingTime = true
}) => {
  // Destructure stats for easier usage
  const { characterCount, wordCount, readingTime } = stats;

  // Convert readingTime (in minutes) to MM:SS format
  const minutes = Math.floor(readingTime);
  const seconds = Math.round((readingTime - minutes) * 60);
  const formattedTime = `${minutes}:${seconds.toString().padStart(2, '0')}`;

  return (
    <div className="space-y-2 p-4 border border-gray-300 rounded-lg">
      <div>
        <strong>Characters:</strong> {characterCount}
      </div>
      <div>
        <strong>Words:</strong> {wordCount}
      </div>
      {showReadingTime && (
        <div>
          <strong>Reading Time:</strong> {formattedTime}
        </div>
      )}
    </div>
  );
};
```

**9. Implement CharacterCounter component and integrate others**

*Explanation:*  
The `CharacterCounter` component will manage the text state, compute statistics, and render `TextInput` and `StatsDisplay`. We’ll use React hooks for state and effects.

**Action:** Open `src/components/CharacterCounter/CharacterCounter.tsx` and add:

```tsx
// src/components/CharacterCounter/CharacterCounter.tsx
import React, { useState, useEffect } from 'react';
import { TextInput } from '../TextInput/TextInput';
import { StatsDisplay } from '../StatsDisplay/StatsDisplay';
import { TextStats, CharacterCounterProps } from '../../types';

/**
 * CharacterCounter parent component
 * Manages text state, computes stats, and renders children.
 */
export const CharacterCounter: React.FC<CharacterCounterProps> = ({
  minWords = 0,
  maxWords = Infinity,
  targetReadingTime = 0
}) => {
  const [text, setText] = useState('');         // holds current text
  const [stats, setStats] = useState<TextStats>({
    characterCount: 0,
    wordCount: 0,
    readingTime: 0
  });

  // Compute stats whenever text changes
  useEffect(() => {
    const characterCount = text.length;
    const wordCount = text.trim() === '' 
      ? 0 
      : text.trim().split(/\s+/).length;
    const readingTime = wordCount / 200; // assuming 200 WPM

    setStats({ characterCount, wordCount, readingTime });
  }, [text]);

  return (
    <div className="max-w-xl mx-auto space-y-4">
      <TextInput onTextChange={setText} initialValue={text} />
      <StatsDisplay stats={stats} showReadingTime={true} />
      {/* You can add progress indicators here based on minWords, maxWords, etc. */}
    </div>
  );
};
```

```tsx
// src/components/CharacterCounter/CharacterCounter.tsx
import React, { useState, useEffect } from 'react';
import { TextInput } from '../TextInput/TextInput';
import { StatsDisplay } from '../StatsDisplay/StatsDisplay';
import { TextStats, CharacterCounterProps } from '../../types';

export const CharacterCounter: React.FC<CharacterCounterProps> = ({
  minWords = 0,
  maxWords = Infinity,
  targetReadingTime = 0
}) => {
  const [text, setText] = useState('');
  const [stats, setStats] = useState<TextStats>({
    characterCount: 0,
    wordCount: 0,
    readingTime: 0
  });

  useEffect(() => {
    const charCount = text.length;
    const wordCount = text.trim() ? text.trim().split(/\s+/).length : 0;
    const readingTime = wordCount / 200;
    setStats({ characterCount: charCount, wordCount, readingTime });
  }, [text]);

  const wordProgress = maxWords === Infinity
    ? 0
    : Math.min(100, (stats.wordCount / maxWords) * 100);
  const progressColor = (stats.wordCount < minWords || stats.wordCount > maxWords)
    ? 'bg-red-500'
    : 'bg-green-500';

  let pace = '';
  if (targetReadingTime > 0) {
    if (stats.readingTime > targetReadingTime * 1.2) pace = 'Slow';
    else if (stats.readingTime < targetReadingTime * 0.8) pace = 'Fast';
    else pace = 'Good pace';
  }

  return (
    <div className="max-w-xl mx-auto space-y-4">
      <TextInput onTextChange={setText} initialValue={text} />
      <StatsDisplay stats={stats} showReadingTime />

      {/* Word Count Progress */}
      <div>
        <div className="w-full bg-gray-200 rounded-full h-2">
          <div
            className={`${progressColor} h-2 rounded-full`}
            style={{ width: `${wordProgress}%` }}
          />
        </div>
        <p className="text-sm mt-1">
          {stats.wordCount} / {maxWords === Infinity ? '∞' : maxWords} words
        </p>
      </div>

      {/* Pace Indicator */}
      {pace && (
        <div className="inline-block px-3 py-1 text-sm bg-blue-100 rounded-full">
          {pace}
        </div>
      )}
    </div>
  );
};
```
