    ' Step 2: Delete rows under A5 until "Meeting Room Information"
    With wsNew
        rowStartDelete = 6
        lastRow = .Cells(.Rows.Count, 1).End(xlUp).Row
        Do While .Cells(rowStartDelete, 1).Value <> "Meeting Room Information" _
              And .Cells(rowStartDelete, 1).Value <> "" _
              And rowStartDelete <= lastRow
            rowStartDelete = rowStartDelete + 1
        Loop
        If rowStartDelete > 6 Then
            .Rows("6:" & rowStartDelete - 1).Delete
        End If

        Do While .Cells(rowStartDelete, 1).Value <> "Internet, Guest Rooms, and AV Estimated Cost Information" _

        Loop
        
        rowEndDelete = rowStartDelete
        Do While .Cells(rowEndDelete, 1).Value <> "" And rowEndDelete <= lastRow
            rowEndDelete = rowEndDelete + 1
