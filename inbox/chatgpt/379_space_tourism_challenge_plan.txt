2. **Create `js/main.js` with fetch + render logic**  
   Replace your `js/main.js` content with the following. It will:
   - Fetch `data.json`
   - Toggle mobile nav
   - Detect which page you’re on (by checking for `.crew-page` or `.technology-page`)
   - Render the corresponding cards into the container divs

   ```js
   // js/main.js

   // 1. Fetch the JSON
   async function fetchData() {
     const res = await fetch('data.json');
     if (!res.ok) throw new Error('Failed to load data.json');
     return res.json();
   }

   // 2. Mobile nav toggle
   function initNavToggle() {
     const toggles = document.querySelectorAll('.navbar-toggler');
     const navCollapse = document.getElementById('navbarNav');
     toggles.forEach(btn => {
       btn.addEventListener('click', () => {
         navCollapse.classList.toggle('show');
       });
     });
   }

   // 3. Render Crew
   function renderCrew(crew) {
     const container = document.querySelector('.crew-container');
     crew.forEach(member => {
       const card = document.createElement('article');
       card.className = 'crew-card';
       card.innerHTML = `
         <picture>
           <source srcset="assets/images/${member.images.webp}" type="image/webp">
           <img src="assets/images/${member.images.png}" alt="${member.name}">
         </picture>
         <div class="crew-info">
           <h3 class="crew-role">${member.role}</h3>
           <h2 class="crew-name">${member.name}</h2>
           <p class="crew-bio">${member.bio}</p>
         </div>
       `;
       container.append(card);
     });
   }

   // 4. Render Technology
   function renderTechnology(items) {
     const container = document.querySelector('.technology-container');
     items.forEach(item => {
       const card = document.createElement('article');
       card.className = 'tech-card';
       card.innerHTML = `
         <picture>
           <source media="(min-width: 768px)" srcset="assets/images/${item.images.portrait}">
           <img src="assets/images/${item.images.landscape}" alt="${item.name}">
         </picture>
         <div class="tech-info">
           <h3 class="tech-term">The Terminology…</h3>
           <h2 class="tech-name">${item.name}</h2>
           <p class="tech-desc">${item.description}</p>
         </div>
       `;
       container.append(card);
     });
   }

   // 5. Init on DOMContentLoaded
   document.addEventListener('DOMContentLoaded', async () => {
     try {
       const data = await fetchData();
       initNavToggle();

       if (document.querySelector('.crew-page')) {
         renderCrew(data.crew);
       }
       if (document.querySelector('.technology-page')) {
         renderTechnology(data.technology);
       }
       // (Later: you can add similar logic for destination pages)
     } catch (err) {
       console.error(err);
     }
   });
   ```
