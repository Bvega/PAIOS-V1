### **Stage 2 – Add simple authentication middleware**

Below is the **complete, updated `server.js`** with plenty of inline comments so you can copy-paste it over your existing file.

```js
// server.js
// -----------------------------------------------------------------------------
// Express demo server showcasing two concepts:
//
//  1. Global request logger      (Activity 1)
//  2. Simple API-key middleware  (Activity 2)
//
//  Public routes  -> no key required
//  Protected /api -> must send header  x-api-key: supersecretkey
// -----------------------------------------------------------------------------

const express = require('express');
const app = express();

// Change the port via an env var or default to 3000
const PORT = process.env.PORT || 3000;

/* ============================================================================
   Activity 1 ▸ GLOBAL REQUEST LOGGER
   Logs: ISO timestamp · HTTP method · path
============================================================================ */
app.use((req, _res, next) => {
  console.log(`[${new Date().toISOString()}] ${req.method} ${req.path}`);
  next();               // hand off to next middleware / route handler
});

/* ============================================================================
   Activity 2 ▸ SIMPLE AUTHENTICATION MIDDLEWARE
   – Only routes under `/api` will use this check.
   – Expected header:  x-api-key: supersecretkey
============================================================================ */
function checkApiKey(req, res, next) {
  const apiKey = req.headers['x-api-key'];     // header names are lowercase
  if (apiKey === 'supersecretkey') {
    return next();                             // ✅ proceed if key is correct
  }
  // ❌ key missing or wrong → deny access
  return res
    .status(401)
    .json({ error: 'Unauthorized: API Key required' });
}

// ✨ Attach the middleware to *just* the /api path space
app.use('/api', checkApiKey);

/* ============================================================================
   PUBLIC ROUTES (no API key required)
============================================================================ */
app.get('/', (_req, res) => {
  res.send('Welcome to the public endpoint!');
});

app.get('/products/:productId', (req, res) => {
  const { productId } = req.params;
  res.send(`Fetching data for product with the required ID: ${productId}`);
});

/* ============================================================================
   PROTECTED ROUTES (require API key)
============================================================================ */
app.get('/api', (_req, res) => {
  res.json({ message: 'Access granted to protected data' });
});

/* ============================================================================
   BOOT SERVER
============================================================================ */
app.listen(PORT, () =>
  console.log(`Server running at http://localhost:${PORT}`)
);
```
