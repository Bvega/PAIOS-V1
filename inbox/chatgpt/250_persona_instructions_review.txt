```js
import { useState, useMemo, useCallback } from 'react';

/**
 * usePagination manages client-side pagination logic.
 *
 * @param {number} totalItems   Total number of items to paginate
 * @param {number} itemsPerPage Items per page (default: 10)
 * @param {number} initialPage  Starting page (default: 1)
 * @returns {{
 *   currentPage: number,
 *   totalPages: number,
 *   startIndex: number,
 *   endIndex: number,
 *   itemsOnCurrentPage: number,
 *   setPage: (page: number) => void,
 *   nextPage: () => void,
 *   prevPage: () => void,
 *   canNextPage: boolean,
 *   canPrevPage: boolean
 * }}
 */
function usePagination(totalItems, itemsPerPage = 10, initialPage = 1) {
  const [currentPage, setCurrentPage] = useState(initialPage);

  const totalPages = useMemo(
    () => Math.max(1, Math.ceil(totalItems / itemsPerPage)),
    [totalItems, itemsPerPage]
  );

  const startIndex = (currentPage - 1) * itemsPerPage;
  const endIndex = Math.min(startIndex + itemsPerPage - 1, totalItems - 1);

  const itemsOnCurrentPage = useMemo(
    () => endIndex - startIndex + 1,
    [startIndex, endIndex]
  );

  const canPrevPage = currentPage > 1;
  const canNextPage = currentPage < totalPages;

  const setPage = useCallback(page => {
    const p = Math.min(Math.max(1, page), totalPages);
    setCurrentPage(p);
  }, [totalPages]);

  const prevPage = useCallback(() => {
    if (canPrevPage) setCurrentPage(p => p - 1);
  }, [canPrevPage]);

  const nextPage = useCallback(() => {
    if (canNextPage) setCurrentPage(p => p + 1);
  }, [canNextPage]);

  return {
    currentPage, totalPages,
    startIndex, endIndex, itemsOnCurrentPage,
    setPage, prevPage, nextPage,
    canPrevPage, canNextPage
  };
}

export default usePagination;
```

**Next Task: Design `useDebounce` Hook API and Outline Behavior**

---

**Design Summary**  
Create a hook that delays updating a value until the specified delay has elapsed without changes.  

- **Signature**  
  ```js
  function useDebounce(value, delay = 500)
  ```
- **Inputs**  
  - `value` (any): the rapidly changing value (e.g., search input)  
  - `delay` (number, ms): debounce interval, defaults to 500  
- **Return**  
  - `debouncedValue` (same type as `value`): only updates after `delay` ms have passed without `value` changing  

- **Behavior**  
  1. Initialize `debouncedValue` state to `value`.  
  2. On every `value` or `delay` change, start a timer (`setTimeout`) for `delay` ms.  
  3. If before the timer fires `value` changes again, clear the previous timer and set a new one.  
  4. When the timer finally fires, update `debouncedValue` to the latest `value`.  

---

**Full Copy-and-Paste Implementation**  
Create `src/hooks/useDebounce.js`:

```js
import { useState, useEffect } from 'react';

/**
 * useDebounce delays updating a value until after a specified delay.
 *
 * @param {any} value        The input value to debounce
 * @param {number} delay     Debounce delay in milliseconds (default: 500)
 * @returns {any}            The debounced value
 */
function useDebounce(value, delay = 500) {
  const [debouncedValue, setDebouncedValue] = useState(value);

  useEffect(() => {
    // Set up a timer to update debouncedValue after the delay
    const timerId = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    // Cleanup: clear timer if value or delay changes before firing
    return () => {
      clearTimeout(timerId);
    };
  }, [value, delay]);

  return debouncedValue;
}

export default useDebounce;
```

```js
// File: src/hooks/__tests__/usePagination.test.js

import { renderHook, act } from '@testing-library/react-hooks';
import usePagination from '../usePagination';

describe('usePagination hook', () => {
  it('initializes correctly with defaults', () => {
    const { result } = renderHook(() => usePagination(42));
    const {
      currentPage,
      totalPages,
      startIndex,
      endIndex,
      itemsOnCurrentPage,
      canPrevPage,
      canNextPage
    } = result.current;

    expect(currentPage).toBe(1);
    expect(totalPages).toBe(Math.ceil(42 / 10)); // default 10/page
    expect(startIndex).toBe(0);
    expect(endIndex).toBe(9);
    expect(itemsOnCurrentPage).toBe(10);
    expect(canPrevPage).toBe(false);
    expect(canNextPage).toBe(true);
  });

  it('nextPage and prevPage work and respect bounds', () => {
    const { result } = renderHook(() => usePagination(25, 5, 2));
    // pages: 1–5, starting at 2
    expect(result.current.currentPage).toBe(2);

    act(() => {
      result.current.nextPage();
    });
    expect(result.current.currentPage).toBe(3);

    act(() => {
      result.current.prevPage();
      result.current.prevPage();
      result.current.prevPage();
    });
    // should not go below 1
    expect(result.current.currentPage).toBe(1);
  });

  it('setPage clamps out-of-range values', () => {
    const { result } = renderHook(() => usePagination(30, 10, 1));
    act(() => {
      result.current.setPage(100);
    });
    expect(result.current.currentPage).toBe(3); // last page
    act(() => {
      result.current.setPage(-5);
    });
    expect(result.current.currentPage).toBe(1);
  });

  it('handles zero items edge case', () => {
    const { result } = renderHook(() => usePagination(0, 10, 1));
    expect(result.current.totalPages).toBe(1);
    expect(result.current.itemsOnCurrentPage).toBe(0);
    expect(result.current.canNextPage).toBe(false);
    expect(result.current.canPrevPage).toBe(false);
  });
});
```

**How it works**  
- `renderHook` mounts the hook in isolation.  
- `act()` wraps any state-changing calls (`nextPage()`, `setPage()`).  
- Assertions verify the API’s returned values and boundary behavior.

```js
import { useState, useMemo, useCallback } from 'react';

function usePagination(totalItems, itemsPerPage = 10, initialPage = 1) {
  const [currentPage, setCurrentPage] = useState(initialPage);

  const totalPages = useMemo(
    () => Math.max(1, Math.ceil(totalItems / itemsPerPage)),
    [totalItems, itemsPerPage]
  );

  const startIndex = useMemo(
    () => (currentPage - 1) * itemsPerPage,
    [currentPage, itemsPerPage]
  );

  const endIndex = useMemo(
    () => Math.min(startIndex + itemsPerPage - 1, totalItems - 1),
    [startIndex, totalItems]
  );

  const itemsOnCurrentPage = useMemo(
    () => Math.max(0, endIndex - startIndex + 1),
    [startIndex, endIndex]
  );

  const canPrevPage = currentPage > 1;
  const canNextPage = currentPage < totalPages;

  const setPage = useCallback(page => {
    const p = Math.min(Math.max(1, page), totalPages);
    setCurrentPage(p);
  }, [totalPages]);

  const prevPage = useCallback(() => {
    if (canPrevPage) setCurrentPage(p => p - 1);
  }, [canPrevPage]);

  const nextPage = useCallback(() => {
    if (canNextPage) setCurrentPage(p => p + 1);
  }, [canNextPage]);

  return {
    currentPage,
    totalPages,
    startIndex,
    endIndex,
    itemsOnCurrentPage,
    setPage,
    prevPage,
    nextPage,
    canPrevPage,
    canNextPage
  };
}

export default usePagination;
```

ASSISTANT:
The failure comes from using a stale `canPrevPage` closure: after multiple `prevPage()` calls in one batch, the hook never updates `canPrevPage` inside that callback, so the page can go below 1 (to 0).

**Fix:** Make `prevPage` and `nextPage` always clamp against bounds inside their updater functions, not by checking a stale boolean. Here’s the corrected full hook:

```js
// src/hooks/usePagination.js
import { useState, useMemo, useCallback } from 'react';

/**
 * usePagination manages client-side pagination logic.
 *
 * @param {number} totalItems   Total number of items to paginate
 * @param {number} itemsPerPage Items per page (default: 10)
 * @param {number} initialPage  Starting page (default: 1)
 * @returns {{
 *   currentPage: number,
 *   totalPages: number,
 *   startIndex: number,
 *   endIndex: number,
 *   itemsOnCurrentPage: number,
 *   setPage: (page: number) => void,
 *   nextPage: () => void,
 *   prevPage: () => void,
 *   canNextPage: boolean,
 *   canPrevPage: boolean
 * }}
 */
function usePagination(totalItems, itemsPerPage = 10, initialPage = 1) {
  const [currentPage, setCurrentPage] = useState(initialPage);

  const totalPages = useMemo(
    () => Math.max(1, Math.ceil(totalItems / itemsPerPage)),
    [totalItems, itemsPerPage]
  );

  const startIndex = useMemo(
    () => (currentPage - 1) * itemsPerPage,
    [currentPage, itemsPerPage]
  );

  const endIndex = useMemo(
    () => Math.min(startIndex + itemsPerPage - 1, totalItems - 1),
    [startIndex, totalItems, itemsPerPage]
  );

  const itemsOnCurrentPage = useMemo(
    () => Math.max(0, endIndex - startIndex + 1),
    [startIndex, endIndex]
  );

  const canPrevPage = currentPage > 1;
  const canNextPage = currentPage < totalPages;

  const setPage = useCallback(
    page => {
      setCurrentPage(p => {
        const next = Math.min(Math.max(1, page), totalPages);
        return next;
      });
    },
    [totalPages]
  );

  const prevPage = useCallback(() => {
    setCurrentPage(p => Math.max(1, p - 1));
  }, []);

  const nextPage = useCallback(() => {
    setCurrentPage(p => Math.min(totalPages, p + 1));
  }, [totalPages]);

  return {
    currentPage,
    totalPages,
    startIndex,
    endIndex,
    itemsOnCurrentPage,
    setPage,
    prevPage,
    nextPage,
    canPrevPage,
    canNextPage
  };
}

export default usePagination;
```

**Next Task: Implement `useDebounce` Hook**

Save this as **`src/hooks/useDebounce.js`**:

```js
import { useState, useEffect } from 'react';

/**
 * useDebounce delays updating a value until a specified delay has passed without changes.
 *
 * @param {any} value    The value to debounce
 * @param {number} delay Debounce delay in milliseconds (default: 500)
 * @returns {any}        The debounced value
 */
function useDebounce(value, delay = 500) {
  const [debouncedValue, setDebouncedValue] = useState(value);

  useEffect(() => {
    // Set up timer to update debouncedValue after delay
    const handler = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    // Cleanup if value or delay changes before timer fires
    return () => {
      clearTimeout(handler);
    };
  }, [value, delay]);

  return debouncedValue;
}

export default useDebounce;
```

**Full Copy-and-Paste for `src/hooks/useDebounce.js`**

```js
import { useState, useEffect } from 'react';

/**
 * useDebounce delays updating a value until a specified delay has passed without changes.
 *
 * @param {any} value    The value to debounce
 * @param {number} delay Debounce delay in milliseconds (default: 500)
 * @returns {any}        The debounced value
 */
function useDebounce(value, delay = 500) {
  const [debouncedValue, setDebouncedValue] = useState(value);

  useEffect(() => {
    const handler = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    return () => clearTimeout(handler);
  }, [value, delay]);

  return debouncedValue;
}

export default useDebounce;
```
