Edit
Here's the AudioPlayer.jsx code:

javascript
import React, { useState, useEffect } from 'react';
import { Play, Pause, SkipBack, SkipForward, Volume2 } from 'lucide-react';

const AudioPlayer = () => {
  const [songs, setSongs] = useState([]);
  const [currentSong, setCurrentSong] = useState(null);
  const [isPlaying, setIsPlaying] = useState(false);
  const [audio] = useState(new Audio());

  useEffect(() => {
    fetchSongs();
  }, []);

  const fetchSongs = async () => {
    try {
      const response = await fetch('http://localhost:3000/api/songs');
      const data = await response.json();
      setSongs(data.songs);
    } catch (error) {
      console.error('Error fetching songs:', error);
    }
  };

  const playSong = (song) => {
    audio.src = `http://localhost:3000/api/stream/${song}`;
    audio.play();
    setCurrentSong(song);
    setIsPlaying(true);
  };

  const togglePlay = () => {
    if (isPlaying) {
      audio.pause();
    } else {
      audio.play();
    }
    setIsPlaying(!isPlaying);
  };

  return (
    <div className="w-full max-w-2xl mx-auto p-4">
      <div className="bg-gray-800 rounded-lg p-4">
        <div className="text-white mb-4">
          <h2 className="text-xl font-bold">{currentSong || 'No song selected'}</h2>
        </div>
        <div className="flex justify-center space-x-4">
          <button className="text-white"><SkipBack /></button>
          <button onClick={togglePlay} className="text-white">
            {isPlaying ? <Pause /> : <Play />}
          </button>
          <button className="text-white"><SkipForward /></button>
          <button className="text-white"><Volume2 /></button>
        </div>
      </div>
      <div className="mt-4">
        <h3 className="text-lg font-bold mb-2">Songs</h3>
        <ul className="space-y-2">
          {songs.map((song, index) => (
            <li 
              key={index}
              onClick={() => playSong(song)}
              className="cursor-pointer hover:bg-gray-100 p-2 rounded"
            >
              {song}
            </li>
          ))}
        </ul>

import React, { useState, useEffect, useRef } from 'react';
import { Play, Pause, SkipBack, SkipForward, Volume2, VolumeX } from 'lucide-react';

const AudioPlayer = () => {
 const [songs, setSongs] = useState([]);
 const [currentSong, setCurrentSong] = useState(null);
 const [currentIndex, setCurrentIndex] = useState(0);
 const [isPlaying, setIsPlaying] = useState(false);
 const [volume, setVolume] = useState(1);
 const [progress, setProgress] = useState(0);
 const [duration, setDuration] = useState(0);
 const audioRef = useRef(new Audio());

 useEffect(() => {
   fetchSongs();
   const audio = audioRef.current;
   
   audio.addEventListener('timeupdate', updateProgress);
   audio.addEventListener('loadedmetadata', () => setDuration(audio.duration));
   audio.addEventListener('ended', handleNext);

   return () => {
     audio.removeEventListener('timeupdate', updateProgress);
     audio.removeEventListener('loadedmetadata', () => setDuration(audio.duration));
     audio.removeEventListener('ended', handleNext);
   };
 }, []);

 const fetchSongs = async () => {
   try {
     const response = await fetch('http://localhost:3000/api/songs');
     const data = await response.json();
     setSongs(data.songs);
   } catch (error) {
     console.error('Error fetching songs:', error);
   }
 };

 const updateProgress = () => {
   const audio = audioRef.current;
   setProgress((audio.currentTime / audio.duration) * 100);
 };

 const handleProgressChange = (e) => {
   const audio = audioRef.current;
   const newTime = (e.target.value / 100) * audio.duration;
   audio.currentTime = newTime;
   setProgress(e.target.value);
 };

   audioRef.current.src = `http://localhost:3000/api/stream/${song}`;
   audioRef.current.play();
   setCurrentSong(song);
   setCurrentIndex(index);
   setIsPlaying(true);
 };

 const togglePlay = () => {
   if (isPlaying) {
     audioRef.current.pause();
   } else {
     audioRef.current.play();
   }
   setIsPlaying(!isPlaying);
 };

   const nextIndex = (currentIndex + 1) % songs.length;
   playSong(songs[nextIndex], nextIndex);
 };

   const prevIndex = (currentIndex - 1 + songs.length) % songs.length;
   playSong(songs[prevIndex], prevIndex);
 };

   audioRef.current.volume = newVolume;
   setVolume(newVolume);
 };

   audioRef.current.volume = newVolume;
   setVolume(newVolume);
 };

   return `${minutes}:${seconds.toString().padStart(2, '0')}`;
 };

   <div className="w-full max-w-2xl mx-auto p-4">
     <div className="bg-gray-800 rounded-lg p-4">

       </div>

           onMouseUp={() => {
             if (isPlaying) audioRef.current.play();
           }}
           className="w-full h-2 bg-gray-200 rounded-lg appearance-none cursor-pointer"
           step="0.1"
           min="0"
           max="100"
         />

       </div>

       {/* Controls */}
       <div className="flex justify-center space-x-4">
         <button onClick={handlePrev} className="text-white"><SkipBack /></button>
         <button onClick={togglePlay} className="text-white">
           {isPlaying ? <Pause /> : <Play />}
         </button>
         <button onClick={handleNext} className="text-white"><SkipForward /></button>
         <div className="flex items-center">
           <button onClick={toggleMute} className="text-white">

           </button>

             max="1"

             onChange={handleVolumeChange}

       </div>
     </div>

     {/* Playlist */}

Here's the complete updated AudioPlayer.jsx code:

javascript
import React, { useState, useEffect, useRef } from 'react';
import { Play, Pause, SkipBack, SkipForward, Volume2, VolumeX } from 'lucide-react';

const AudioPlayer = () => {
  const [songs, setSongs] = useState([]);
  const [currentSong, setCurrentSong] = useState(null);
  const [currentIndex, setCurrentIndex] = useState(0);
  const [isPlaying, setIsPlaying] = useState(false);
  const [volume, setVolume] = useState(1);
  const [progress, setProgress] = useState(0);
  const [duration, setDuration] = useState(0);
  const audioRef = useRef(new Audio());

  useEffect(() => {
    fetchSongs();
    const audio = audioRef.current;
    
    audio.addEventListener('timeupdate', updateProgress);
    audio.addEventListener('loadedmetadata', () => setDuration(audio.duration));
    audio.addEventListener('ended', handleNext);

    return () => {
      audio.removeEventListener('timeupdate', updateProgress);
      audio.removeEventListener('loadedmetadata', () => setDuration(audio.duration));
      audio.removeEventListener('ended', handleNext);
    };
  }, []);

  const fetchSongs = async () => {
    try {
      console.log('Fetching songs...'); // Debug log 1
      const response = await fetch('http://localhost:3000/api/songs');
      const data = await response.json();
      console.log('Songs data received:', data); // Debug log 2
      console.log('Setting songs state:', data.songs); // Debug log 3
      setSongs(data.songs);
      console.log('Current songs state:', songs); // Debug log 4
    } catch (error) {
      console.error('Error fetching songs:', error);
    }
  };

  const updateProgress = () => {
    const audio = audioRef.current;
    setProgress((audio.currentTime / audio.duration) * 100);
  };

  const handleProgressChange = (e) => {
    const audio = audioRef.current;
    const newTime = (e.target.value / 100) * audio.duration;
    audio.currentTime = newTime;
    setProgress(e.target.value);
  };

  const playSong = (song, index) => {
    audioRef.current.src = `http://localhost:3000/api/stream/${song}`;
    audioRef.current.play();
    setCurrentSong(song);
    setCurrentIndex(index);
    setIsPlaying(true);
  };

  const togglePlay = () => {
    if (isPlaying) {
      audioRef.current.pause();
    } else {
      audioRef.current.play();
    }
    setIsPlaying(!isPlaying);
  };

  const handleNext = () => {
    const nextIndex = (currentIndex + 1) % songs.length;
    playSong(songs[nextIndex], nextIndex);
  };

  const handlePrev = () => {
    const prevIndex = (currentIndex - 1 + songs.length) % songs.length;
    playSong(songs[prevIndex], prevIndex);
  };

  const handleVolumeChange = (e) => {
    const newVolume = parseFloat(e.target.value);
    audioRef.current.volume = newVolume;
    setVolume(newVolume);
  };

  const toggleMute = () => {
    const newVolume = volume === 0 ? 1 : 0;
    audioRef.current.volume = newVolume;
    setVolume(newVolume);
  };

  const formatTime = (time) => {
    if (isNaN(time)) return '0:00';
    const minutes = Math.floor(time / 60);
    const seconds = Math.floor(time % 60);
    return `${minutes}:${seconds.toString().padStart(2, '0')}`;
  };

  return (
    <div className="w-full max-w-2xl mx-auto p-4">
      <div className="bg-gray-800 rounded-lg p-4">
        {/* Album Art */}
        <div className="flex justify-center mb-4">
          <img 
            src={currentSong 
              ? `http://localhost:3000/api/album-art/${currentSong.replace('.mp3', '.jpg')}` 
              : `http://localhost:3000/api/album-art/default.jpg`
            }
            alt="Album Art"
            className="w-48 h-48 rounded-lg shadow-lg"
            onError={(e) => {
              e.target.src = `http://localhost:3000/api/album-art/default.jpg`;
            }}
          />
        </div>

        <div className="text-white mb-4">
          <h2 className="text-xl font-bold text-center">{currentSong || 'No song selected'}</h2>
        </div>
        
        {/* Progress Bar */}
        <div className="mb-4">
          <input 
            type="range" 
            value={progress}
            onChange={handleProgressChange}
            onMouseDown={() => audioRef.current.pause()}
            onMouseUp={() => {
              if (isPlaying) audioRef.current.play();
            }}
            className="w-full h-2 bg-gray-200 rounded-lg appearance-none cursor-pointer"
            step="0.1"
            min="0"
            max="100"
          />
          <div className="flex justify-between text-white text-sm">
            <span>{formatTime(duration * (progress / 100))}</span>
            <span>{formatTime(duration)}</span>
          </div>
        </div>

        {/* Controls */}
        <div className="flex justify-center space-x-4">
          <button onClick={handlePrev} className="text-white hover:text-gray-300"><SkipBack /></button>
          <button onClick={togglePlay} className="text-white hover:text-gray-300">
            {isPlaying ? <Pause /> : <Play />}
          </button>
          <button onClick={handleNext} className="text-white hover:text-gray-300"><SkipForward /></button>
          <div className="flex items-center">
            <button onClick={toggleMute} className="text-white hover:text-gray-300">
              {volume === 0 ? <VolumeX /> : <Volume2 />}
            </button>
            <input 
              type="range" 
              min="0" 
              max="1" 
              step="0.1" 
              value={volume}
              onChange={handleVolumeChange}
              className="w-20 ml-2"
            />
          </div>
        </div>
      </div>

      {/* Playlist */}
      <div className="mt-4">
        <h3 className="text-lg font-bold mb-2">Playlist ({songs.length} songs)</h3>
        <ul className="space-y-2">
          {console.log('Rendering songs:', songs)} {/* Debug log 5 */}
          {songs && songs.map((song, index) => (
            <li 
              key={index}
              onClick={() => playSong(song, index)}
              className={`cursor-pointer p-2 rounded ${
                currentSong === song 
                  ? 'bg-gray-800 text-white' 
                  : 'hover:bg-gray-100'
              }`}
            >
              {song}
            </li>
          ))}
        </ul>
      </div>
    </div>
  );
};

export default AudioPlayer;

## 1. Create `server/index.js`

In **server/index.js**, delete anything there now and replace it with:

```js
import express from 'express'
import cors from 'cors'
import { fileURLToPath } from 'url'
import { dirname, join } from 'path'
import fs from 'fs'
import { promises as fsp } from 'fs'

const __filename = fileURLToPath(import.meta.url)
const __dirname  = dirname(__filename)
const app        = express()
const PORT       = process.env.PORT || 3000

app.use(cors())
app.use(express.json())

// 1. Health check
app.get('/health', (_, res) => {
  res.json({ status: 'ok' })
})

// 2. List all .mp3 files
app.get('/api/songs', (req, res) => {
  const musicDir = join(__dirname, '../music_files')
  fs.readdir(musicDir, (err, files) => {
    if (err) return res.status(500).json({ error: 'Cannot read music folder' })
    const mp3s = files.filter(f => f.endsWith('.mp3'))
    res.json({ songs: mp3s })
  })
})

// 3. Stream audio with support for range requests
app.get('/api/stream/:filename', (req, res) => {
  const filePath = join(__dirname, '../music_files', req.params.filename)
  const stat     = fs.statSync(filePath)
  const fileSize = stat.size
  const range    = req.headers.range

  if (range) {
    const [startStr, endStr] = range.replace(/bytes=/, '').split('-')
    const start = parseInt(startStr, 10)
    const end   = endStr ? parseInt(endStr, 10) : fileSize - 1
    const chunkSize = (end - start) + 1

    res.writeHead(206, {
      'Content-Range' : `bytes ${start}-${end}/${fileSize}`,
      'Accept-Ranges': 'bytes',
      'Content-Length': chunkSize,
      'Content-Type' : 'audio/mpeg'
    })

    fs.createReadStream(filePath, { start, end }).pipe(res)
  } else {
    res.writeHead(200, {
      'Content-Length': fileSize,
      'Content-Type'  : 'audio/mpeg'
    })
    fs.createReadStream(filePath).pipe(res)
  }
})

// 4. Serve album art (fallback to default.jpg)
app.get('/api/album-art/:filename', async (req, res) => {
  const artDir      = join(__dirname, '../music_files/album-art')
  const artPath     = join(artDir, req.params.filename)
  const defaultArt  = join(artDir, 'default.jpg')

  try {
    await fsp.access(artPath)
    res.sendFile(artPath)
  } catch {
    res.sendFile(defaultArt)
  }
})

app.listen(PORT, () => {
  console.log(`🎵 ÁMBAR Radio Server is running on http://localhost:${PORT}`)
})
```

   import cors from 'cors'
   import { fileURLToPath } from 'url'
   import { dirname, join } from 'path'
   import fs from 'fs'
   import { promises as fsp } from 'fs'

   const __filename = fileURLToPath(import.meta.url)
   const __dirname = dirname(__filename)
   const app = express()
   const PORT = process.env.PORT || 3000

   app.use(cors())
   app.use(express.json())

   app.get('/health', (_, res) => res.json({ status: 'ok' }))

   app.get('/api/songs', (req, res) => {
     const dir = join(__dirname, '../music_files')
     fs.readdir(dir, (e,f) => e ? res.status(500).json({ error: e.message }) : res.json({ songs: f.filter(s=>s.endsWith('.mp3')) }))

   app.get('/api/stream/:file', (req, res) => {
     const pathFile = join(__dirname,'../music_files',req.params.file)
     const stat = fs.statSync(pathFile)
     const range = req.headers.range
     if (range) {
       const [start,end] = range.replace(/bytes=/,'').split('-')
       const st = +start, en = end?+end:stat.size-1
       res.writeHead(206,{
         'Content-Range':`bytes ${st}-${en}/${stat.size}`,
         'Accept-Ranges':'bytes',
         'Content-Length':en-st+1,
         'Content-Type':'audio/mpeg'
       })
       fs.createReadStream(pathFile,{start:st,end:en}).pipe(res)
     } else {

     const art = join(__dirname,'../music_files/album-art',req.params.file)
     const def = join(__dirname,'../music_files/album-art/default.jpg')
     try { await fsp.access(art); res.sendFile(art) }
     catch { res.sendFile(def) }
   })
