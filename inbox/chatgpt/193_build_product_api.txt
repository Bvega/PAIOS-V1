const productSchema = new mongoose.Schema(
  {
    name: {
      type: String,
      required: [true, 'Product name is required'],
      trim: true,
    },
    price: {
      type: Number,
      required: [true, 'Price is required'],
      min: [0, 'Price cannot be negative'],
    },

      type: String,
      default: 'uncategorized',
      trim: true,
    },
  },
  { timestamps: true }
);

**`src/routes/product.routes.js`**
```js
import { Router } from 'express';
import Product from '../models/Product.js';

const router = Router();

/**
 * POST /products
 * Creates a new product
 */
router.post('/', async (req, res) => {
  try {
    const newProduct = await Product.create(req.body);
    return res.status(201).json(newProduct);
  } catch (err) {
    return res.status(400).json({ error: err.message });
  }
});

/**
 * GET /products
 * Returns all products
 */
router.get('/', async (_req, res) => {
  try {
    const products = await Product.find({});
    return res.json(products);
  } catch (err) {
    return res.status(500).json({ error: 'Server error' });
  }
});

/**
 * DELETE /products/:id
 * Deletes a product by its MongoDB _id
 */
router.delete('/:id', async (req, res) => {
  const { id } = req.params;
  try {
    const deleted = await Product.findByIdAndDelete(id);
    if (!deleted) {
      return res.status(404).json({ error: 'Product not found' });
    }
    return res.json({ message: 'Product deleted successfully' });
  } catch (err) {
    return res.status(400).json({ error: 'Invalid ID format' });
  }
});

export default router;
```

**`tests/product.test.js`**
```js
/**
 * Integration tests for POST, GET, DELETE /products
 *
 * Uses:
 * - mongodb-memory-server to host an ephemeral MongoDB
 * - Supertest to perform HTTP assertions
 */
import request from 'supertest';
import mongoose from 'mongoose';
import { MongoMemoryServer } from 'mongodb-memory-server';

import app from '../src/server.js';
import Product from '../src/models/Product.js';

let mongo;

/* Spin up in-memory Mongo before any tests run */
beforeAll(async () => {
  mongo = await MongoMemoryServer.create();
  const uri = mongo.getUri();
  await mongoose.connect(uri, {
    useNewUrlParser: true,
    useUnifiedTopology: true,
  });
});

/* Clear data after each test to keep isolation */
afterEach(async () => {
  await Product.deleteMany({});
});

/* Shut everything down once all tests finish */
afterAll(async () => {
  await mongoose.disconnect();
  await mongo.stop();
});

describe('Product API', () => {
  it('POST /products should create a product', async () => {
    const res = await request(app)
      .post('/products')
      .send({ name: 'Pen', price: 1.99, category: 'stationery' });

    expect(res.statusCode).toBe(201);
    expect(res.body).toMatchObject({
      name: 'Pen',
      price: 1.99,
      category: 'stationery',
    });
  });

  it('GET /products should return list containing created product', async () => {
    await Product.create({ name: 'Notebook', price: 3.5 });
    const res = await request(app).get('/products');

    expect(res.statusCode).toBe(200);
    expect(res.body.length).toBe(1);
    expect(res.body[0].name).toBe('Notebook');
  });

  it('DELETE /products/:id should remove the product', async () => {
    const prod = await Product.create({ name: 'Eraser', price: 0.5 });
    const res = await request(app).delete(`/products/${prod._id}`);

    expect(res.statusCode).toBe(200);
    expect(res.body).toEqual({ message: 'Product deleted successfully' });

    const exists = await Product.findById(prod._id);
    expect(exists).toBeNull();
  });
});
```
