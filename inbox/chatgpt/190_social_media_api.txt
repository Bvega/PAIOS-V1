```js
/* -----------------------------------------------
   server.js
   Simple Express API that connects to MongoDB Atlas
------------------------------------------------ */

require("dotenv").config();            // 🔒 Load variables from .env
const express = require("express");    // 🚀 Minimal web framework
const { MongoClient } = require("mongodb");

const app  = express();
const PORT = process.env.PORT || 3001; // Allow overriding in prod

// ---- MongoDB connection URI (from .env) ----
const uri = process.env.MONGO_URI;     // e.g. mongodb+srv://user:pass@cluster0.xxxx.mongodb.net/?retryWrites=true&w=majority

/* -----------------------------------------------
   Helper: connectToDatabase()
   1. Creates a single shared client.
   2. Attempts initial connection at startup so the
      / route can report success/failure.
------------------------------------------------ */
const client = new MongoClient(uri, {
  useNewUrlParser: true,
  useUnifiedTopology: true,
});

async function connectToDatabase() {
  try {
    // Only create one connection for the whole app
    if (!client.topology?.isConnected()) {
      await client.connect();
      console.log("✅  MongoDB connected");
    }
    return true;
  } catch (err) {
    console.error("❌  MongoDB connection error:", err.message);
    return false;
  }
}

/* -----------------------------------------------
   Routes
------------------------------------------------ */
// Root route – indicates DB connection status
app.get("/", async (_req, res) => {
  const isConnected = await connectToDatabase();
  if (isConnected) {
    res.json({ message: "Successfully connected to the database!" });
  } else {
    res.status(500).json({ message: "Failed to connect to the database." });
  }
});

/* -----------------------------------------------
   Start server
------------------------------------------------ */
app.listen(PORT, async () => {
  // Attempt initial DB connection on boot
  await connectToDatabase();
  console.log(`🚀  Server running at http://localhost:${PORT}`);
});
```

```js
  try {
    if (!client.topology?.isConnected()) {
      await client.connect();
      console.log("✅  MongoDB connected");
    }
    return true;

    console.error("❌  MongoDB connection error:", err.message);
    return false;
  }
}

  const isConnected = await connectToDatabase();
  if (isConnected) {
    res.json({ message: "Successfully connected to the database!" });
  } else {
    res.status(500).json({ message: "Failed to connect to the database." });
  }
});
```
