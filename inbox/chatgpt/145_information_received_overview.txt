Sub CreateMeetingRoomInformationSheet()

    Dim W As Workbook
    Dim wsSource As Worksheet, wsMeeting As Worksheet
    Dim rowToStartDeleting As Long, rowToDelete As Long
    Dim startRow As Long, endRow As Long

    Set W = ActiveWorkbook
    Set wsSource = W.Sheets("Submitted proposal")

    ' Copy directly after Submitted proposal, then rename
    wsSource.Copy After:=wsSource
    Set wsMeeting = W.ActiveSheet
    On Error Resume Next
    wsMeeting.Name = "Meeting Rooms"
    On Error GoTo 0

    ' Step 2: delete A6..row before "Meeting Room Information"
    With wsMeeting
        rowToStartDeleting = 6
        Do While .Cells(rowToStartDeleting, 1).Value <> "Meeting Room Information" _
            And .Cells(rowToStartDeleting, 1).Value <> "" _
            And rowToStartDeleting <= .Rows.Count
            rowToStartDeleting = rowToStartDeleting + 1
        Loop
        If .Cells(rowToStartDeleting, 1).Value = "Meeting Room Information" Then
            If rowToStartDeleting > 6 Then .Rows("6:" & rowToStartDeleting - 1).Delete
        Else
            ' section not found -> skip next block
            GoTo SkipMeetingSection
        End If

        ' Steps 3&4: delete from "Internet..." down to next blank
        rowToDelete = rowToStartDeleting
        Do While .Cells(rowToDelete, 1).Value <> "Internet, Guest Rooms, and AV Estimated Cost Information" _
            And .Cells(rowToDelete, 1).Value <> "" _
            And rowToDelete <= .Rows.Count
            rowToDelete = rowToDelete + 1
        Loop
        If .Cells(rowToDelete, 1).Value = "Internet, Guest Rooms, and AV Estimated Cost Information" Then
            Dim bottomRow As Long
            bottomRow = rowToDelete
            Do While .Cells(bottomRow, 1).Value <> "" And bottomRow <= .Rows.Count
                bottomRow = bottomRow + 1
            Loop
            .Rows(rowToDelete & ":" & bottomRow - 1).Delete
        End If
    End With

SkipMeetingSection:

    ' Step 5: remove the section in the source sheet
    With wsSource
        startRow = 1
        Do While .Cells(startRow, 1).Value <> "Meeting Room Information" _
            And startRow <= .Rows.Count
            startRow = startRow + 1
        Loop
        If .Cells(startRow, 1).Value <> "Meeting Room Information" Then Exit Sub

        endRow = startRow
        Do While .Cells(endRow, 1).Value <> "Internet, Guest Rooms, and AV Estimated Cost Information" _
            And endRow <= .Rows.Count
            endRow = endRow + 1
        Loop
        If .Cells(endRow, 1).Value = "Internet, Guest Rooms, and AV Estimated Cost Information" Then
            .Rows(startRow & ":" & endRow - 1).Delete
        End If
    End With
End Sub

Private Function SheetExistsInWB(ByVal wb As Workbook, ByVal sheetName As String) As Boolean
    Dim sh As Object
    SheetExistsInWB = False
    For Each sh In wb.Sheets
        If sh.Name = sheetName Then SheetExistsInWB = True: Exit Function
    Next sh
End Function

Private Function CleanFileName(ByVal s As String) As String
    Dim bad(): bad = Array("\", "/", ":", "*", "?", """", "<", ">", "|")
    Dim i As Long
    For i = LBound(bad) To UBound(bad)
        s = Replace(s, bad(i), "_")
    Next
    CleanFileName = s
End Function

Private Function GetBaseNameNoExt(ByVal fileName As String) As String
    ' strips .xlsx or .xlsm (and any other 4 or 5 char extension)
    Dim p As Long
    p = InStrRev(fileName, ".")
    If p > 0 Then
        GetBaseNameNoExt = Left(fileName, p - 1)
    Else
        GetBaseNameNoExt = fileName
    End If
End Function

Private Function GetFolderFromPath(ByVal p As String) As String
    'Ensures trailing backslash and returns folder part
    Dim s As String: s = p
    If Len(s) = 0 Then GetFolderFromPath = "": Exit Function
    If Right$(s, 1) <> "\" Then s = s & "\"
    GetFolderFromPath = s
End Function

Private Function GetBaseNameNoExt(ByVal fileName As String) As String
    Dim p As Long
    p = InStrRev(fileName, ".")
    If p > 0 Then
        GetBaseNameNoExt = Left$(fileName, p - 1)
    Else
        GetBaseNameNoExt = fileName
    End If
End Function

Private Function CleanFileName(ByVal s As String) As String
    'remove characters invalid in Windows filenames
    Dim bad As Variant, i As Long
    bad = Array("<", ">", ":", """", "/", "\", "|", "?", "*")
    For i = LBound(bad) To UBound(bad)
        s = Replace$(s, bad(i), "_")
    Next
    'normalize double spaces and trim
    Do While InStr(s, "  ") > 0
        s = Replace$(s, "  ", " ")
    Loop
    CleanFileName = Trim$(s)
End Function

Private Function SheetExists(wb As Workbook, sheetName As String) As Boolean
    Dim sh As Object
    SheetExists = False
    For Each sh In wb.Sheets

Private Function CleanFileName(ByVal s As String) As String
    Dim bad As Variant, i As Long
    bad = Array("<", ">", ":", """", "/", "\", "|", "?", "*")
    For i = LBound(bad) To UBound(bad)
        s = Replace$(s, bad(i), "_")
    Next
    Do While InStr(s, "  ") > 0
        s = Replace$(s, "  ", " ")
    Loop
    CleanFileName = Trim$(s)
End Function

Sub CreateMeetingRoomInformationSheet()

    Dim wsSource As Worksheet
    Dim wsMeeting As Worksheet
    Dim row_to_start_deleting As Long
    Dim row_to_delete As Long
    Dim start_row As Long
    Dim end_row As Long

    ' Step 1: Create a copy of "Submitted proposal" and rename to "Meeting Rooms"
    Set wsSource = ActiveWorkbook.Sheets("Submitted proposal")

    ' If "Meeting Rooms" already exists, delete it first to avoid duplicates
    On Error Resume Next
    ActiveWorkbook.Sheets("Meeting Rooms").Delete
    On Error GoTo 0

    wsSource.Copy After:=wsSource
    Set wsMeeting = ActiveSheet
    wsMeeting.Name = "Meeting Rooms"

    ' Step 2: On "Meeting Rooms" delete rows below A5 until "Meeting Room Information"
    With wsMeeting
        row_to_start_deleting = 6
        Do While .Cells(row_to_start_deleting, 1).Value <> "Meeting Room Information" _
             And .Cells(row_to_start_deleting, 1).Value <> "" _
             And row_to_start_deleting <= .Rows.Count
            row_to_start_deleting = row_to_start_deleting + 1
        Loop
        If row_to_start_deleting > 6 Then
            .Rows("6:" & row_to_start_deleting - 1).Delete
        End If

        ' Step 3 & 4: Delete from "Internet..." down to the next blank cell (if found)
        Do While .Cells(row_to_start_deleting, 1).Value <> "Internet, Guest Rooms, and AV Estimated Cost Information" _
             And .Cells(row_to_start_deleting, 1).Value <> "" _
             And row_to_start_deleting <= .Rows.Count
            row_to_start_deleting = row_to_start_deleting + 1
        Loop

        If .Cells(row_to_start_deleting, 1).Value = "Internet, Guest Rooms, and AV Estimated Cost Information" Then
            row_to_delete = row_to_start_deleting
            Do While .Cells(row_to_delete, 1).Value <> "" And row_to_delete <= .Rows.Count
                row_to_delete = row_to_delete + 1
            Loop
            .Rows(row_to_start_deleting & ":" & row_to_delete - 1).Delete
        End If
    End With

    ' Step 5: In "Submitted proposal", delete rows from "Meeting Room Information" up to (but not including) "Internet..."
    With wsSource
        start_row = 1
        Do While .Cells(start_row, 1).Value <> "Meeting Room Information" And start_row <= .Rows.Count
            start_row = start_row + 1
        Loop
        If .Cells(start_row, 1).Value <> "Meeting Room Information" Then
            ' marker not found; skip this step gracefully
            Exit Sub
        End If

        end_row = start_row
        Do While .Cells(end_row, 1).Value <> "Internet, Guest Rooms, and AV Estimated Cost Information" And end_row <= .Rows.Count
            end_row = end_row + 1
        Loop
        If .Cells(end_row, 1).Value = "Internet, Guest Rooms, and AV Estimated Cost Information" Then
            .Rows(start_row & ":" & end_row - 1).Delete
        End If
    End With

End Sub
