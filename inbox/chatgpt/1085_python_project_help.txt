#### 3. `csv_handler.py`
This module handles all CSV operations, such as loading the passwords into memory, saving updates, and ensuring the data integrity.

```python
import csv

class CSVHandler:
    def __init__(self, filename='passwords.csv'):
        self.filename = filename

    def load_passwords(self):
        passwords = []
        with open(self.filename, newline='') as csvfile:
            reader = csv.DictReader(csvfile)
            for row in reader:
                passwords.append(row)
        return passwords

    def save_passwords(self, passwords):
        with open(self.filename, 'w', newline='') as csvfile:
            fieldnames = ['username', 'password', 'service']
            writer = csv.DictWriter(csvfile, fieldnames=fieldnames)
            writer.writeheader()
            for password in passwords:
                writer.writerow(password)
```

#### 4. `password_manager.py`
This module will include the core logic for adding, deleting, and retrieving passwords. It will interact with the `CSVHandler` to persist data.

```python
from csv_handler import CSVHandler

class PasswordManager:
    def __init__(self):
        self.csv_handler = CSVHandler()
        self.passwords = self.csv_handler.load_passwords()

    def add_password(self, username, password, service):
        new_entry = {'username': username, 'password': password, 'service': service}
        self.passwords.append(new_entry)
        self.csv_handler.save_passwords(self.passwords)

    def delete_password(self, service):
        self.passwords = [p for p in self.passwords if p['service'] != service]
        self.csv_handler.save_passwords(self.passwords)

    def get_password(self, service):
        for p in self.passwords:
            if p['service'] == service:
                return p
        return None
```

from password_manager import PasswordManager

class TestPasswordManager(unittest.TestCase):

        self.password_manager = PasswordManager(self.test_file)

    def tearDown(self):
        os.remove(self.test_file)

    def test_add_and_get_password(self):

        result = self.password_manager.get_password('service1')
        expected = {'username': 'user1', 'password': 'pass1', 'service': 'service1'}
        self.assertEqual(result, expected)

    def test_delete_password(self):

        self.password_manager.delete_password('service1')
