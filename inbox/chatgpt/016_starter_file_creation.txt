import argparse, re, sys

try:

try:
    import docx  # python-docx
except Exception:

ROOT = Path(__file__).resolve().parents[1]

INPUT = ROOT / "input"
SNAPSHOTS = ROOT / "snapshots"

def slugify(s: str) -> str:
    s = s.strip().lower()
    s = re.sub(r"[^\w\s-]", "", s)
    s = re.sub(r"[\s_-]+", "-", s)
    s = re.sub(r"^-+|-+$", "", s)
    return s or "item"

def extract_text(path: Path) -> str:
    ext = path.suffix.lower()
    if ext in [".txt", ".md"]:
        return path.read_text(encoding="utf-8", errors="ignore")
    if ext == ".pdf":
        if PyPDF2 is None:
            raise RuntimeError("PyPDF2 not installed. Install it with: pip install PyPDF2")
        text = []
        with open(path, "rb") as f:
            reader = PyPDF2.PdfReader(f)
            for i, page in enumerate(reader.pages):
                try:
                    text.append(page.extract_text() or "")
                except Exception:
                    text.append("")
        return "\n".join(text)
    if ext in [".docx"]:
        if docx is None:
            raise RuntimeError("python-docx not installed. Install it with: pip install python-docx")
        d = docx.Document(str(path))
        return "\n".join([p.text for p in d.paragraphs])
    raise RuntimeError(f"Unsupported file type: {ext} (supported: .pdf, .docx, .txt, .md)")

def guess_title(text: str, fallback: str) -> str:
    # Use first non-empty line up to 120 chars as title
    for line in text.splitlines():
        line = line.strip()
        if len(line) >= 5:
            return line[:120]
    return fallback
