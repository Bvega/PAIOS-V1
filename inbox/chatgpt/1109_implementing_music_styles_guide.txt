def blend_styles(styles, selected_styles):
    """
    Blend multiple music styles to create a unique style.

    Args:
    styles (list): List of available styles.
    selected_styles (list): List of styles to blend.

    Returns:
    dict: A dictionary representing the blended style characteristics.
    """
    blended_style = {
        "style": " & ".join(selected_styles),
        "instrumentation": [],
        "vocals": [],
        "tempo": "Varied",
        "mood": "Mixed"
    }
    
    for style in styles:
        if style["style"] in selected_styles:
            blended_style["instrumentation"].extend(style["instrumentation"])
            blended_style["vocals"].append(style["vocals"])
            blended_style["mood"] = style["mood"]  # Use logic to blend moods

    # Deduplicate instrumentation and vocals
    blended_style["instrumentation"] = list(set(blended_style["instrumentation"]))
    blended_style["vocals"] = " & ".join(set(blended_style["vocals"]))

    return blended_style
