### ⚙️ **JavaScript (`script.js`)**
```javascript
document.addEventListener("DOMContentLoaded", () => {
  const taskInput = document.getElementById("taskInput");
  const addTaskButton = document.getElementById("addTaskButton");
  const taskList = document.getElementById("taskList");

  let tasks = JSON.parse(localStorage.getItem("tasks")) || [];

  const saveTasks = () => {
    localStorage.setItem("tasks", JSON.stringify(tasks));
  };

  const renderTasks = () => {
    taskList.innerHTML = "";
    tasks.forEach((task, index) => {
      const li = document.createElement("li");
      li.className = "flex items-center justify-between bg-gray-100 px-4 py-2 rounded";

      const text = document.createElement("span");
      text.textContent = task.text;
      text.className = task.done ? "line-through text-gray-500" : "";
      text.classList.add("cursor-pointer", "flex-1");
      text.onclick = () => {
        tasks[index].done = !tasks[index].done;
        saveTasks();
        renderTasks();
      };

      const delBtn = document.createElement("button");
      delBtn.textContent = "❌";
      delBtn.className = "ml-2 text-red-500 hover:text-red-700";
      delBtn.onclick = () => {
        tasks.splice(index, 1);
        saveTasks();
        renderTasks();
      };

      li.append(text, delBtn);
      taskList.appendChild(li);
    });
  };

  addTaskButton.onclick = () => {
    const value = taskInput.value.trim();
    if (value) {
      tasks.push({ text: value, done: false });
      taskInput.value = "";
      saveTasks();
      renderTasks();
    }
  };

  renderTasks();
});
```

## ⚙️ **JavaScript (`script.js`) with Comments**

```javascript
// Wait for the DOM to load before running any JavaScript
document.addEventListener("DOMContentLoaded", () => {
  // Get references to HTML elements
  const taskInput = document.getElementById("taskInput");
  const addTaskButton = document.getElementById("addTaskButton");
  const taskList = document.getElementById("taskList");

  // Load existing tasks from local storage, or start with an empty list
  let tasks = JSON.parse(localStorage.getItem("tasks")) || [];

  // Save current tasks to local storage so they're remembered on refresh
  const saveTasks = () => {
    localStorage.setItem("tasks", JSON.stringify(tasks));
  };

  // Render the task list in the UI
  const renderTasks = () => {
    // Clear current list display
    taskList.innerHTML = "";

    // Loop through all tasks and create list items
    tasks.forEach((task, index) => {
      const li = document.createElement("li");
      li.className = "flex items-center justify-between bg-gray-100 px-4 py-2 rounded";

      // Create a text element that shows the task
      const text = document.createElement("span");
      text.textContent = task.text;
      text.className = task.done ? "line-through text-gray-500" : "";
      text.classList.add("cursor-pointer", "flex-1");

      // Toggle "done" state when the text is clicked
      text.onclick = () => {
        tasks[index].done = !tasks[index].done;
        saveTasks();
        renderTasks();
      };

      // Create a delete button for removing tasks
      const delBtn = document.createElement("button");
      delBtn.textContent = "❌";
      delBtn.className = "ml-2 text-red-500 hover:text-red-700";

      // When clicked, remove the task
      delBtn.onclick = () => {
        tasks.splice(index, 1); // remove one task at position 'index'
        saveTasks();
        renderTasks();
      };

      // Add the task text and delete button to the list item
      li.append(text, delBtn);

      // Add the list item to the task list
      taskList.appendChild(li);
    });
  };

  // Add new task when the "Add" button is clicked
  addTaskButton.onclick = () => {
    const value = taskInput.value.trim();

    // Prevent empty tasks from being added
    if (value) {
      tasks.push({ text: value, done: false }); // add new task to list
      taskInput.value = ""; // clear input field
      saveTasks(); // update local storage
      renderTasks(); // update UI
    }
  };

  // Initial load of saved tasks
  renderTasks();
});
```

```javascript
// Wait until the page loads to run any JavaScript
document.addEventListener("DOMContentLoaded", () => {
  // Get references to DOM elements
  const taskInput = document.getElementById("taskInput");
  const addTaskButton = document.getElementById("addTaskButton");
  const taskList = document.getElementById("taskList");

  // Load tasks from localStorage or use an empty array
  let tasks = JSON.parse(localStorage.getItem("tasks")) || [];

  // Save tasks to localStorage
  const saveTasks = () => {
    localStorage.setItem("tasks", JSON.stringify(tasks));
  };

  // Render the entire task list
  const renderTasks = () => {
    taskList.innerHTML = ""; // Clear existing list

    tasks.forEach((task, index) => {
      // Create a list item for each task
      const li = document.createElement("li");
      li.className = "flex items-center justify-between bg-gray-100 px-4 py-2 rounded";

      // Create a wrapper for the icon and task text or input
      const textWrapper = document.createElement("div");
      textWrapper.className = "flex items-center flex-1 gap-2";

      // Create the SVG icon (checkbox)
      const icon = document.createElementNS("http://www.w3.org/2000/svg", "svg");
      icon.setAttribute("viewBox", "0 0 24 24");
      icon.setAttribute("width", "24");
      icon.setAttribute("height", "24");

      const path = document.createElementNS("http://www.w3.org/2000/svg", "path");

      if (task.done) {
        // Green checkmark if completed
        path.setAttribute("fill", "#10B981");
        path.setAttribute("d", "M9 16.17L4.83 12l-1.42 1.41L9 19 21 7l-1.41-1.41z");
      } else {
        // Empty gray square if not completed
        path.setAttribute("fill", "none");
        path.setAttribute("stroke", "#9CA3AF");
        path.setAttribute("stroke-width", "2");
        path.setAttribute("d", "M5 5h14v14H5z");
      }

      icon.appendChild(path);

      // Check if task is in editing mode
      if (task.editing) {
        // Input field to edit task text
        const input = document.createElement("input");
        input.type = "text";
        input.value = task.text;
        input.className = "flex-1 border border-blue-300 px-2 py-1 rounded";

        // Save on Enter key
        input.onkeydown = (e) => {
          if (e.key === "Enter") {
            task.text = input.value.trim() || task.text;
            task.editing = false;
            saveTasks();
            renderTasks();
          }
        };

        // Save on click-away (blur)
        input.onblur = () => {
          task.text = input.value.trim() || task.text;
          task.editing = false;
          saveTasks();
          renderTasks();
        };

        textWrapper.appendChild(icon);
        textWrapper.appendChild(input);
      } else {
        // Static task text view
        const text = document.createElement("span");
        text.textContent = task.text;
        text.className = task.done ? "line-through text-gray-500" : "";
        text.classList.add("flex-1", "cursor-pointer");

        // Double-click to edit
        text.ondblclick = () => {
          task.editing = true;
          renderTasks();
        };

        // Click once to toggle "done" state
        text.onclick = () => {
          tasks[index].done = !tasks[index].done;
          saveTasks();
          renderTasks();
        };

        textWrapper.appendChild(icon);
        textWrapper.appendChild(text);
      }

      // Create delete button
      const delBtn = document.createElement("button");
      delBtn.textContent = "❌";
      delBtn.className = "ml-2 text-red-500 hover:text-red-700";
      delBtn.onclick = () => {
        tasks.splice(index, 1);
        saveTasks();
        renderTasks();
      };

      // Add to list item and append to list
      li.appendChild(textWrapper);
      li.appendChild(delBtn);
      taskList.appendChild(li);
    });
  };

  // Add task when "Add" button is clicked
  addTaskButton.onclick = () => {
    const value = taskInput.value.trim();
    if (value) {
      // Add new task with default states
      tasks.push({ text: value, done: false, editing: false });
      taskInput.value = ""; // Clear input
      saveTasks(); // Save to localStorage
      renderTasks(); // Update UI
    }
  };

  // Initial display
  renderTasks();
});
```
