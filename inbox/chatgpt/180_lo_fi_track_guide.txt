## 🛠️ Option B: Generate Locally with Python

If you'd like to create or modify the files yourself, here’s a simple Python script using the open-source library **music21**. Just install the library and run the script:

```python
from music21 import stream, chord, note, tempo, meter, instrument, metadata

# Settings
bpm = 70
ts = meter.TimeSignature('4/4')

# Create parts for piano, bass, and flute
piano = stream.Part()
piano.append(instrument.Piano())
piano.append(tempo.MetronomeMark(number=bpm))
piano.append(ts)

bass = stream.Part()
bass.append(instrument.ElectricBass())
bass.append(tempo.MetronomeMark(number=bpm))
bass.append(ts)

flute = stream.Part()
flute.append(instrument.Flute())
flute.append(tempo.MetronomeMark(number=bpm))
flute.append(ts)

# Chord progression: Cm7 – Abmaj7 – Bb7 – Gm7
chords = [
    ["C4", "Eb4", "G4", "Bb4"],
    ["Ab3", "C4", "Eb4", "G4"],
    ["Bb3", "D4", "F4", "Ab4"],
    ["G3", "Bb3", "D4", "F4"],
]

# Append two iterations (8 bars total)
for _ in range(2):
    for chord_notes in chords:
        ch = chord.Chord(chord_notes)
        ch.duration.quarterLength = 4
        piano.append(ch)
        # Bass note
        bass_note = note.Note(chord_notes[0].replace("4","2"))
        bass_note.duration.quarterLength = 4
        bass.append(bass_note)

# Flute melody motif
melody = [
    ("G4",1), ("rest",1), ("Bb4",2),
    ("C5",1), ("Eb5",1), ("rest",2),
    ("D5",2), ("G4",2)
]
for _ in range(2):
    for pitch, dur in melody:
        if pitch == "rest":
            fl = note.Rest(quarterLength=dur)
        else:
            fl = note.Note(pitch, quarterLength=dur)
        flute.append(fl)

# Combine into score and save
score = stream.Score()
score.insert(0, metadata.Metadata(title="Chill Lo‑Fi Loop", composer="Music Composer GPT"))
score.append([piano, bass, flute])
score.write("midi", fp="chill_lofi_loop.mid")
score.write("musicxml", fp="chill_lofi_loop.xml")
```
