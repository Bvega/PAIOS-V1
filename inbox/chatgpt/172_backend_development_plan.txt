ASSISTANT:
**Hygen** and **Plop** are code‑generation tools that scaffold files from templates so you don’t hand‑write the same boilerplate over and over.

---

## Hygen (fast, file‑system–driven scaffolder)
- **What it is:** A zero‑config CLI that runs “generators” defined by folders and template files in a special `_templates/` directory.
- **How it works:** Each generator/action lives in a folder (e.g., `_templates/model/new/`). Templates use **EJS** syntax and a small YAML front‑matter to declare where to write, and whether to inject into existing files.
- **Best for:** Very fast, repeatable stamping of files (models, controllers, routes) with minimal ceremony. Great when you already know the inputs and want **low‑interaction, high speed**.
- **Notable features:**  
  - File **injection** (append/insert into existing files).  
  - Works well in teams by committing `_templates/` to the repo.  
  - Simple param passing: `hygen model new --name Project`.

---

## Plop (interactive micro‑generator framework)
- **What it is:** A small framework that defines generators in a **`plopfile`** (JavaScript/TypeScript). You script prompts and actions programmatically.
- **How it works:** Prompts (via Inquirer) collect answers, then **Handlebars** templates and “actions” (add/modify/append) create or edit files.
- **Best for:** **Interactive flows** and **complex logic** (conditional files, multiple steps, custom actions). Fits when you want guided Q&A for consistent inputs.
- **Notable features:**  
  - Rich prompts & validation.  
  - Powerful **modify/append** actions to inject code (e.g., add a route import to an index).  
  - Extensible with helpers and custom actions.

---

## When to choose which
- **Speed & simplicity, repo‑native templates → Hygen.**  
  You drop templates in `_templates/` and go. Ideal for CRUD scaffolds you’ll run many times in one day.
- **Complex logic, interactive wizards, conditional outputs → Plop.**  
  Ideal when you want prompts (e.g., “Add auth? y/n”) and smart branching that changes files across the codebase.

---

## Quick mental model
- **Hygen:** “Folders are the config.” Run with flags. EJS templates + YAML front‑matter decide destination and injections.  
- **Plop:** “Code is the config.” A `plopfile` defines prompts and actions. Handlebars templates render, and actions can add/modify/append anywhere.

const mongoose = require('mongoose');

async function connectDB(uri) {
  if (!uri) {
    console.error('❌ MONGO_URI is missing. Set it in your .env file.');
    process.exit(1);
  }

  mongoose.connection.on('connected', () => {
    console.log('🗄️  MongoDB connected');
  });

  mongoose.connection.on('error', (err) => {
    console.error('❌ MongoDB connection error:', err.message);
  });

  mongoose.connection.on('disconnected', () => {
    console.warn('⚠️  MongoDB disconnected');
  });

  // Retry-friendly options
  const opts = {
    autoIndex: true,
    maxPoolSize: 10
  };

  await mongoose.connect(uri, opts);
  return mongoose.connection;
}

// Graceful shutdown
process.on('SIGINT', async () => {
  await mongoose.connection.close();
  console.log('🔌 MongoDB connection closed due to app termination');
  process.exit(0);
});

module.exports = connectDB;
```

// Minimal JWT utilities: sign and verify tokens.
// Stores the user id in `sub` (subject) claim.

const jwt = require('jsonwebtoken');

function ensureSecret() {
  const secret = process.env.JWT_SECRET;
  if (!secret) throw new Error('JWT_SECRET is missing. Set it in your .env file.');
  return secret;
}

function signToken(userId, options = {}) {
  const secret = ensureSecret();
  const payload = { sub: userId };
  const jwtOptions = { expiresIn: options.expiresIn || '7d' };
  return jwt.sign(payload, secret, jwtOptions);
}

function verifyToken(token) {
  const secret = ensureSecret();
  return jwt.verify(token, secret);
}

module.exports = { signToken, verifyToken };

### `models/User.js`

// User schema: unique email, hashed password (bcrypt), helper to compare passwords.

const mongoose = require('mongoose');
const bcrypt = require('bcrypt');

const userSchema = new mongoose.Schema(
  {
    username: {
      type: String,
      required: true,
      trim: true,
      minlength: 2,
      maxlength: 64
    },
    email: {
      type: String,
      required: true,
      unique: true,
      lowercase: true,
      trim: true,
      match: [/^\S+@\S+\.\S+$/, 'Invalid email format']
    },
    password: {
      type: String,
      required: true,
      minlength: 8,
      select: false // exclude by default
    }
  },
  { timestamps: true }
);

// Hash password before save when it’s modified
userSchema.pre('save', async function (next) {
  if (!this.isModified('password')) return next();
  const saltRounds = 10;
  this.password = await bcrypt.hash(this.password, saltRounds);
  next();
});

// Compare candidate password to stored hash
userSchema.methods.comparePassword = function (candidate) {
  return bcrypt.compare(candidate, this.password);
};

// Hide password and __v on JSON output
userSchema.methods.toJSON = function () {
  const obj = this.toObject({ getters: true });
  delete obj.password;
  delete obj.__v;
  return obj;
};

module.exports = mongoose.model('User', userSchema);
```

### `controllers/authController.js`
```js
// Auth controllers: register and login.
// Keep them thin; validation minimal for the student MVP.

const User = require('../models/User');
const asyncHandler = require('../utils/asyncHandler');
const { signToken } = require('../utils/jwt');

// POST /api/users/register
exports.register = asyncHandler(async (req, res) => {
  const { username, email, password } = req.body;

  if (!username || !email || !password) {
    return res.status(400).json({ error: 'BadRequest', message: 'username, email, and password are required' });
  }

  // Friendly unique email check
  const existing = await User.findOne({ email });
  if (existing) {
    return res.status(409).json({ error: 'Conflict', message: 'Email already in use' });
  }

  const user = await User.create({ username, email, password });
  const token = signToken(user._id.toString());
  res.status(201).json({ user, token });
});

// POST /api/users/login
exports.login = asyncHandler(async (req, res) => {
  const { email, password } = req.body;
  if (!email || !password) {
    return res.status(400).json({ error: 'BadRequest', message: 'email and password are required' });
  }

  // Select password explicitly because we set select:false on schema
  const user = await User.findOne({ email }).select('+password');
  if (!user) {
    return res.status(401).json({ error: 'Unauthorized', message: 'Invalid credentials' });
  }

  const match = await user.comparePassword(password);
  if (!match) {
    return res.status(401).json({ error: 'Unauthorized', message: 'Invalid credentials' });
  }

  const token = signToken(user._id.toString());
  const safeUser = user.toObject();
  delete safeUser.password;
  res.json({ user: safeUser, token });
});
```

### `middleware/auth.js`
```js
// JWT auth middleware. Reads Authorization: Bearer <token> and attaches req.user = { id }.

const { verifyToken } = require('../utils/jwt');

module.exports = function authenticate(req, res, next) {
  const header = req.headers['authorization'] || '';
  const [scheme, token] = header.split(' ');

  if (scheme !== 'Bearer' || !token) {
    return res.status(401).json({ error: 'Unauthorized', message: 'Missing Bearer token' });
  }

  try {
    const decoded = verifyToken(token);
    req.user = { id: decoded.sub };
    next();
  } catch (_err) {
    return res.status(401).json({ error: 'Unauthorized', message: 'Invalid or expired token' });
  }
};
```

// Ownership helpers for Project-based resources.
// Use loadProjectAndAssertOwner for routes with :id.
// Use assertProjectOwner(projectId, userId) for nested resources (e.g., Tasks).

const asyncHandler = require('../utils/asyncHandler');
const Project = require('../models/Project');

// Generic helper: throws 403 if user is not owner
async function assertProjectOwner(projectId, userId) {
  const project = await Project.findById(projectId);
  if (!project) {
    const err = new Error('Project not found');
    err.status = 404;
    throw err;
  }
  if (project.user.toString() !== userId.toString()) {
    const err = new Error('Forbidden: not the project owner');
    err.status = 403;
    throw err;
  }
  return project;
}

// Middleware: loads project by req.params.id and asserts ownership; attaches as req.project
const loadProjectAndAssertOwner = asyncHandler(async (req, _res, next) => {
  const projectId = req.params.id;
  const userId = req.user?.id;
  const project = await assertProjectOwner(projectId, userId);
  req.project = project;
  next();
});

module.exports = { assertProjectOwner, loadProjectAndAssertOwner };
