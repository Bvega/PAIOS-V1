class TemplateGeneratorApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Template Generator")
        
        self.setup_menu()
        self.setup_ui()
        self.load_config()
        
        logging.basicConfig(filename='template_generator.log', level=logging.INFO)

    def setup_menu(self):
        menubar = tk.Menu(self.root)
        file_menu = tk.Menu(menubar, tearoff=0)
        file_menu.add_command(label="New", command=self.new_template)
        file_menu.add_command(label="Save", command=self.save_template)
        file_menu.add_separator()
        file_menu.add_command(label="Exit", command=self.root.quit)
        menubar.add_cascade(label="File", menu=file_menu)
        self.root.config(menu=menubar)

    def setup_ui(self):
        self.generate_button = tk.Button(self.root, text="Generate Template", command=self.generate_template)
        self.generate_button.pack(pady=10)

        self.output_area = scrolledtext.ScrolledText(self.root, wrap=tk.WORD, width=100, height=30)
        self.output_area.pack(padx=10, pady=10)

        self.status_bar = tk.Label(self.root, text="Ready", bd=1, relief=tk.SUNKEN, anchor=tk.W)
        self.status_bar.pack(side=tk.BOTTOM, fill=tk.X)

    def load_config(self):
        try:
            with open('config.json', 'r') as f:
                self.config = json.load(f)
        except FileNotFoundError:
            self.config = {}  # Use default values if config file not found

    def generate_template(self):
        try:
            generated_template = generate_variation()
            self.output_area.delete("1.0", tk.END)
            self.output_area.insert(tk.END, generated_template)
            self.status_bar.config(text="Template generated successfully")
            logging.info("Template generated")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {str(e)}")
            logging.error(f"Error generating template: {str(e)}")

    def new_template(self):
        self.output_area.delete("1.0", tk.END)
        self.status_bar.config(text="New template")

    def save_template(self):
        template = self.output_area.get("1.0", tk.END)
        file_path = filedialog.asksaveasfilename(defaultextension=".txt")
        if file_path:
            with open(file_path, 'w') as f:
                f.write(template)
            self.status_bar.config(text=f"Template saved to {file_path}")
            logging.info(f"Template saved to {file_path}")

if __name__ == "__main__":
