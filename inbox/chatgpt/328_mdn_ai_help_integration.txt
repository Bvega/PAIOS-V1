### **Step 1: HTML Form Structure**

```html
<!-- blog-form.html -->
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8" />
  <meta name="viewport" content="width=device-width, initial-scale=1.0" />
  <title>Blog Submission Form</title>
  <link rel="stylesheet" href="form-style.css" />
</head>
<body>
  <main>
    <h1>Submit a Blog Post</h1>
    <form id="blogForm" novalidate>
      <fieldset>
        <legend>Author Information</legend>

        <label for="name">Name:</label>
        <input type="text" id="name" name="name" required minlength="3" />

        <label for="email">Email:</label>
        <input type="email" id="email" name="email" required />
      </fieldset>

      <fieldset>
        <legend>Blog Content</legend>

        <label for="title">Blog Title:</label>
        <input type="text" id="title" name="title" required minlength="5" />

        <label for="content">Content:</label>
        <textarea id="content" name="content" rows="8" required></textarea>
      </fieldset>

      <button type="submit">Submit</button>
    </form>
    <div id="confirmation" role="status" aria-live="polite"></div>
  </main>

  <script src="form-script.js"></script>
</body>
</html>
```

### **Step 3: JavaScript for Basic Validation**

```javascript
// form-script.js
document.getElementById('blogForm').addEventListener('submit', function (e) {
  e.preventDefault();

  const form = e.target;
  const name = form.name.value.trim();
  const email = form.email.value.trim();
  const title = form.title.value.trim();
  const content = form.content.value.trim();

  if (!name || !email || !title || !content) {
    alert("Please fill in all fields correctly.");
    return;
  }

  // Optional: Show a confirmation message
  document.getElementById('confirmation').textContent = "Your blog post has been submitted!";
  form.reset();
});
```

- Uses `aria-live` for dynamic confirmation message
- Uses external CSS and JS for clean separation of concerns
