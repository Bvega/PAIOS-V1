import React from 'react';
import { StatsDisplayProps } from '../../types';

/**
 * StatsDisplay component shows character count, word count, and reading time.
 */
const StatsDisplay: React.FC<StatsDisplayProps> = ({
  stats,
  showReadingTime = true,
}) => {
  const { characterCount, wordCount, readingTime } = stats;
  const minutes = Math.floor(readingTime);
  const seconds = Math.round((readingTime - minutes) * 60)
    .toString()
    .padStart(2, '0');
  return (
    <div className="grid grid-cols-2 gap-4 mt-4">
      <div>
        <h3 className="text-lg font-medium">Characters</h3>
        <p>{characterCount}</p>
      </div>
      <div>
        <h3 className="text-lg font-medium">Words</h3>
        <p>{wordCount}</p>
      </div>
      {showReadingTime && (
        <div>
          <h3 className="text-lg font-medium">Reading Time</h3>
          <p>{minutes}:{seconds}</p>
        </div>
      )}
    </div>
  );
};

export default StatsDisplay;

**`src/components/CharacterCounter/CharacterCounter.tsx`**
```tsx
import React, { useState, useEffect } from 'react';
import TextInput from '../TextInput/TextInput';
import StatsDisplay from '../StatsDisplay/StatsDisplay';
import { CharacterCounterProps, TextStats } from '../../types';

/**
 * Helper to calculate text statistics.
 */
const calculateStats = (text: string): TextStats => {
  const characterCount = text.length;
  const wordCount = text.trim() === '' ? 0 : text.trim().split(/\s+/).length;
  const readingTime = wordCount / 200; // average 200 WPM
  return { characterCount, wordCount, readingTime };
};

/**
 * CharacterCounter combines TextInput and StatsDisplay, managing state.
 */
const CharacterCounter: React.FC<CharacterCounterProps> = ({
  minWords = 0,
  maxWords = Infinity,
  targetReadingTime = 0,
}) => {
  const [text, setText] = useState('');
  const [stats, setStats] = useState<TextStats>(calculateStats(''));

  useEffect(() => {
    setStats(calculateStats(text));
  }, [text]);

  // Calculate progress bar percentage
  const progress =
    stats.wordCount < minWords
      ? (stats.wordCount / minWords) * 100
      : minWords === Infinity
      ? 0
      : (stats.wordCount / maxWords) * 100;

  return (
    <div>
      <TextInput onTextChange={setText} initialValue={text} />
      <StatsDisplay stats={stats} showReadingTime />
      <div className="mt-4">
        <p>
          Min: {minWords} | Max: {maxWords}
        </p>
        <div className="w-full bg-gray-200 rounded-full h-2 mt-2">
          <div
            className="bg-blue-500 h-2 rounded-full"
            style={{ width: `${Math.min(progress, 100)}%` }}
          />
        </div>
      </div>
    </div>
  );
};

export default CharacterCounter;
```
