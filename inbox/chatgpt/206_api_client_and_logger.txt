### Stage 1 – `apiClient.js`  
Create a shared Axios instance, pre-pointed at the JSONPlaceholder base URL, with a request-logging interceptor.

```js
// apiClient.js
// ------------------------------
// Shared Axios instance for JSONPlaceholder API
// ------------------------------

import axios from "axios";

/**
 * Pre-configured Axios client.
 * - baseURL: every request is relative to this
 * - timeout: fail fast after 10 s
 */
const apiClient = axios.create({
  baseURL: "https://jsonplaceholder.typicode.com",
  timeout: 10_000,
});

/**
 * Request interceptor 👉 logs every outgoing call.
 * Example:  Requesting: GET /users
 */
apiClient.interceptors.request.use((config) => {
  console.log(`Requesting: ${config.method?.toUpperCase()} ${config.url}`);
  return config;            // continue with the request
});

export default apiClient;
```

### **apiClient.js**  (root folder)

```js
/**
 * apiClient.js
 * ------------------------------------------------------------------
 * A shared, pre-configured Axios instance for talking to the
 * JSONPlaceholder API. Centralizing this logic keeps every route
 * cleaner and lets us add cross-cutting concerns (auth, retries,
 * logging, caching) in one place.
 */

import axios from "axios"; // 1️⃣  Bring in Axios

/**
 * 2️⃣  Create the client.
 * - baseURL   → All relative paths ("/users", "/posts") resolve here.
 * - timeout   → Abort if the server takes >10 s to respond.
 */
const apiClient = axios.create({
  baseURL: "https://jsonplaceholder.typicode.com",
  timeout: 10_000, // **10 000 ms = 10 s**
});

/**
 * 3️⃣  Request interceptor:
 * Any time *this* client makes a call, we log the HTTP verb and path.
 *   Example output:  Requesting: GET /users
 */
apiClient.interceptors.request.use((config) => {
  // config.method is lowercase per Axios spec; make it uppercase for readability
  console.log(`Requesting: ${config.method?.toUpperCase()} ${config.url}`);
  return config; // Important ✔️ — pass the config along so the request continues
});

/**
 * 4️⃣  Export the configured instance.
 * Any file can now simply `import apiClient from "../apiClient.js";`
 */
export default apiClient;
```

### **src/server.js**

```js
/**
 * server.js
 * ------------------------------------------------------------------
 * Minimal Express server exposing a `/users` endpoint.
 * The endpoint proxies (fetches) JSONPlaceholder data using the
 * shared Axios client created in apiClient.js.
 */

import express from "express";        // Express web-framework
import apiClient from "../apiClient.js"; // Shared Axios client (note the .. path)

/* ------------------------------------------------------------------
   1️⃣  App setup
------------------------------------------------------------------- */
const app = express();

/* ------------------------------------------------------------------
   2️⃣  Route: GET /users
------------------------------------------------------------------- */
app.get("/users", async (_req, res) => {
  try {
    // 2.1  Forward request to JSONPlaceholder
    const { data } = await apiClient.get("/users");

    // 2.2  Emit the JSON straight back to the caller
    res.json(data);
  } catch (error) {
    // 2.3  Any network/API failure ends up here
    console.error("Error fetching users:", error.message);
    res.status(500).json({ message: "Unable to fetch users" });
  }
});

/* ------------------------------------------------------------------
   3️⃣  Start the HTTP server
------------------------------------------------------------------- */
const PORT = 3000;
app.listen(PORT, () =>
  console.log(`🚀 Server running on http://localhost:${PORT}`)
);
```
